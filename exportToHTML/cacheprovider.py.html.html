        <span class="s3">if </span><span class="s1">isinstance(collector</span><span class="s3">, </span><span class="s1">Module) </span><span class="s3">and not </span><span class="s1">isinstance(collector</span><span class="s3">, </span><span class="s1">Package):</span>

                <span class="s3">return </span><span class="s1">CollectReport(</span>
                    <span class="s1">collector.nodeid</span><span class="s3">, </span><span class="s4">&quot;passed&quot;</span><span class="s3">, </span><span class="s1">longrepr=</span><span class="s3">None, </span><span class="s1">result=[]</span>

    <span class="s3">def </span><span class="s1">pytest_runtest_logreport(self</span><span class="s3">, </span><span class="s1">report: TestReport) -&gt; </span><span class="s3">None</span><span class="s1">:</span>

    <span class="s3">def </span><span class="s1">pytest_collectreport(self</span><span class="s3">, </span><span class="s1">report: CollectReport) -&gt; </span><span class="s3">None</span><span class="s1">:</span>
        <span class="s1">passed = report.outcome </span><span class="s3">in </span><span class="s1">(</span><span class="s4">&quot;passed&quot;</span><span class="s3">, </span><span class="s4">&quot;skipped&quot;</span><span class="s1">)</span>

    <span class="s1">@hookimpl(hookwrapper=</span><span class="s3">True, </span><span class="s1">tryfirst=</span><span class="s3">True</span><span class="s1">)</span>
    <span class="s3">def </span><span class="s1">pytest_collection_modifyitems(</span>
        <span class="s1">self</span><span class="s3">, </span><span class="s1">config: Config</span><span class="s3">, </span><span class="s1">items: List[nodes.Item]</span>
    <span class="s1">) -&gt; Generator[</span><span class="s3">None, None, None</span><span class="s1">]:</span>
        <span class="s3">yield</span>

        <span class="s3">if not </span><span class="s1">self.active:</span>
            <span class="s3">return</span>

        <span class="s3">if </span><span class="s1">self.lastfailed:</span>

            <span class="s1">previously_passed = []</span>
            <span class="s3">for </span><span class="s1">item </span><span class="s3">in </span><span class="s1">items:</span>
                <span class="s3">if </span><span class="s1">item.nodeid </span><span class="s3">in </span><span class="s1">self.lastfailed:</span>
                    <span class="s1">previously_failed.append(item)</span>
                <span class="s3">else</span><span class="s1">:</span>
                    <span class="s1">previously_passed.append(item)</span>
            <span class="s1">self._previously_failed_count = len(previously_failed)</span>

            <span class="s3">if not </span><span class="s1">previously_failed:</span>
                <span class="s2"># Running a subset of all tests with recorded failures</span>
                <span class="s2"># only outside of it.</span>
                <span class="s1">self._report_status = </span><span class="s4">&quot;%d known failures not in selected tests&quot; </span><span class="s1">% (</span>
                    <span class="s1">len(self.lastfailed)</span><span class="s3">,</span>
                <span class="s1">)</span>
            <span class="s3">else</span><span class="s1">:</span>
                <span class="s3">if </span><span class="s1">self.config.getoption(</span><span class="s4">&quot;lf&quot;</span><span class="s1">):</span>
                    <span class="s1">items[:] = previously_failed</span>
                    <span class="s1">config.hook.pytest_deselected(items=previously_passed)</span>
                <span class="s3">else</span><span class="s1">:  </span><span class="s2"># --failedfirst</span>
                    <span class="s1">items[:] = previously_failed + previously_passed</span>

                <span class="s1">noun = </span><span class="s4">&quot;failure&quot; </span><span class="s3">if </span><span class="s1">self._previously_failed_count == </span><span class="s6">1 </span><span class="s3">else </span><span class="s4">&quot;failures&quot;</span>
                <span class="s1">suffix = </span><span class="s4">&quot; first&quot; </span><span class="s3">if </span><span class="s1">self.config.getoption(</span><span class="s4">&quot;failedfirst&quot;</span><span class="s1">) </span><span class="s3">else </span><span class="s4">&quot;&quot;</span>
                <span class="s1">self._report_status = </span><span class="s4">&quot;rerun previous {count} {noun}{suffix}&quot;</span><span class="s1">.format(</span>
                    <span class="s1">count=self._previously_failed_count</span><span class="s3">, </span><span class="s1">suffix=suffix</span><span class="s3">, </span><span class="s1">noun=noun</span>
                <span class="s1">)</span>

            <span class="s3">if </span><span class="s1">self._skipped_files &gt; </span><span class="s6">0</span><span class="s1">:</span>
                <span class="s1">files_noun = </span><span class="s4">&quot;file&quot; </span><span class="s3">if </span><span class="s1">self._skipped_files == </span><span class="s6">1 </span><span class="s3">else </span><span class="s4">&quot;files&quot;</span>
                <span class="s1">self._report_status += </span><span class="s4">&quot; (skipped {files} {files_noun})&quot;</span><span class="s1">.format(</span>
                    <span class="s1">files=self._skipped_files</span><span class="s3">, </span><span class="s1">files_noun=files_noun</span>
                <span class="s1">)</span>
        <span class="s3">else</span><span class="s1">:</span>
            <span class="s1">self._report_status = </span><span class="s4">&quot;no previously failed tests, &quot;</span>
            <span class="s3">if </span><span class="s1">self.config.getoption(</span><span class="s4">&quot;last_failed_no_failures&quot;</span><span class="s1">) == </span><span class="s4">&quot;none&quot;</span><span class="s1">:</span>
                <span class="s1">self._report_status += </span><span class="s4">&quot;deselecting all items.&quot;</span>
                <span class="s1">config.hook.pytest_deselected(items=items[:])</span>
                <span class="s1">items[:] = []</span>
            <span class="s3">else</span><span class="s1">:</span>
                <span class="s1">self._report_status += </span><span class="s4">&quot;not deselecting items.&quot;</span>

    <span class="s3">def </span><span class="s1">pytest_sessionfinish(self</span><span class="s3">, </span><span class="s1">session: Session) -&gt; </span><span class="s3">None</span><span class="s1">:</span>
        <span class="s1">config = self.config</span>
        <span class="s3">if </span><span class="s1">config.getoption(</span><span class="s4">&quot;cacheshow&quot;</span><span class="s1">) </span><span class="s3">or </span><span class="s1">hasattr(config</span><span class="s3">, </span><span class="s4">&quot;workerinput&quot;</span><span class="s1">):</span>
            <span class="s3">return</span>

        <span class="s3">assert </span><span class="s1">config.cache </span><span class="s3">is not None</span>
        <span class="s1">saved_lastfailed = config.cache.get(</span><span class="s4">&quot;cache/lastfailed&quot;</span><span class="s3">, </span><span class="s1">{})</span>

<span class="s3">class </span><span class="s1">NFPlugin:</span>
    <span class="s0">&quot;&quot;&quot;Plugin which implements the --nf (run new-first) option.&quot;&quot;&quot;</span>

    <span class="s3">def </span><span class="s1">__init__(self</span><span class="s3">, </span><span class="s1">config: Config) -&gt; </span><span class="s3">None</span><span class="s1">:</span>

        <span class="s3">assert </span><span class="s1">config.cache </span><span class="s3">is not None</span>
        <span class="s1">self.cached_nodeids = set(config.cache.get(</span><span class="s4">&quot;cache/nodeids&quot;</span><span class="s3">, </span><span class="s1">[]))</span>

    <span class="s1">@hookimpl(hookwrapper=</span><span class="s3">True, </span><span class="s1">tryfirst=</span><span class="s3">True</span><span class="s1">)</span>
    <span class="s3">def </span><span class="s1">pytest_collection_modifyitems(</span>
        <span class="s1">self</span><span class="s3">, </span><span class="s1">items: List[nodes.Item]</span>
    <span class="s1">) -&gt; Generator[</span><span class="s3">None, None, None</span><span class="s1">]:</span>

        <span class="s3">if </span><span class="s1">self.active:</span>
            <span class="s1">new_items: Dict[str</span><span class="s3">, </span><span class="s1">nodes.Item] = {}</span>

                <span class="s1">new_items.values()</span>
            <span class="s1">) + self._get_increasing_order(other_items.values())</span>
            <span class="s1">self.cached_nodeids.update(new_items)</span>
        <span class="s3">else</span><span class="s1">:</span>

    <span class="s3">def </span><span class="s1">pytest_sessionfinish(self) -&gt; </span><span class="s3">None</span><span class="s1">:</span>

            <span class="s3">return</span>

        <span class="s3">if </span><span class="s1">config.getoption(</span><span class="s4">&quot;collectonly&quot;</span><span class="s1">):</span>
            <span class="s3">return</span>
