<span class="s2">def </span><span class="s1">write_installed_files_from_setuptools_record(</span>

<span class="s1">) -&gt; </span><span class="s2">None</span><span class="s1">:</span>

        <span class="s1">message = (</span>
            <span class="s3">&quot;{} did not indicate that it installed an &quot;</span>
            <span class="s3">&quot;.egg-info directory. Only setup.py projects &quot;</span>
            <span class="s3">&quot;generating .egg-info directories are supported.&quot;</span>

    <span class="s1">new_lines = []</span>

                <span class="s1">install_options=install_options</span><span class="s2">,</span>
                <span class="s1">record_filename=record_filename</span><span class="s2">,</span>
                <span class="s1">root=root</span><span class="s2">,</span>
                <span class="s1">prefix=prefix</span><span class="s2">,</span>

                <span class="s2">return False</span>

        <span class="s2">except </span><span class="s1">Exception </span><span class="s2">as </span><span class="s1">e:</span>

        <span class="s4"># At this point, we have successfully installed the requirement.</span>

        <span class="s4"># We intentionally do not use any encoding to read the file because</span>
        <span class="s4"># setuptools writes the file using distutils.file_util.write_file,</span>
        <span class="s4"># which does not specify an encoding.</span>
        <span class="s2">with </span><span class="s1">open(record_filename) </span><span class="s2">as </span><span class="s1">f:</span>
            <span class="s1">record_lines = f.read().splitlines()</span>

    <span class="s1">write_installed_files_from_setuptools_record(record_lines</span><span class="s2">, </span><span class="s1">root</span><span class="s2">, </span><span class="s1">req_description)</span>
    <span class="s2">return True</span>
