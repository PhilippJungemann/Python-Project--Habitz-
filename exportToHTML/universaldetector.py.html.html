    <span class="s2">&quot;&quot;&quot; 
    The ``UniversalDetector`` class underlies the ``chardet.detect`` function 
    and coordinates all of the different charset probers. 
 
    To get a ``dict`` containing an encoding and its confidence, you can simply

    .. code::

            u.feed(some_bytes) 
            u.close() 
            detected = u.result

    <span class="s1">MINIMUM_THRESHOLD = </span><span class="s4">0.20</span>
    <span class="s1">HIGH_BYTE_DETECTOR = re.compile(</span><span class="s5">b&quot;[</span><span class="s3">\x80</span><span class="s5">-</span><span class="s3">\xFF</span><span class="s5">]&quot;</span><span class="s1">)</span>

    <span class="s1">ISO_WIN_MAP = {</span>
        <span class="s6">&quot;iso-8859-1&quot;</span><span class="s1">: </span><span class="s6">&quot;Windows-1252&quot;</span><span class="s3">,</span>

    <span class="s3">def </span><span class="s1">reset(self):</span>
        <span class="s2">&quot;&quot;&quot; 
        Reset the UniversalDetector and all of its probers back to their 
        initial states.  This is called by ``__init__``, so you only need to 
        call this directly in between analyses of different documents. 
        &quot;&quot;&quot;</span>
        <span class="s1">self.result = {</span><span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s3">None, </span><span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">0.0</span><span class="s3">, </span><span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s3">None</span><span class="s1">}</span>
        <span class="s1">self.done = </span><span class="s3">False</span>
        <span class="s1">self._got_data = </span><span class="s3">False</span>
        <span class="s1">self._has_win_bytes = </span><span class="s3">False</span>
        <span class="s1">self._input_state = InputState.PURE_ASCII</span>
        <span class="s1">self._last_char = </span><span class="s5">b&quot;&quot;</span>
        <span class="s3">if </span><span class="s1">self._esc_charset_prober:</span>
            <span class="s1">self._esc_charset_prober.reset()</span>
        <span class="s3">if </span><span class="s1">self._utf1632_prober:</span>
            <span class="s1">self._utf1632_prober.reset()</span>
        <span class="s3">for </span><span class="s1">prober </span><span class="s3">in </span><span class="s1">self._charset_probers:</span>
            <span class="s1">prober.reset()</span>

    <span class="s3">def </span><span class="s1">feed(self</span><span class="s3">, </span><span class="s1">byte_str):</span>
        <span class="s2">&quot;&quot;&quot; 
        Takes a chunk of a document and feeds it through all of the relevant 
        charset probers. 
 
        After calling ``feed``, you can check the value of the ``done`` 
        attribute to see if you need to continue feeding the 
        ``UniversalDetector`` more data, or if it has made a prediction 
        (in the ``result`` attribute). 
 
        .. note:: 
           You should always call ``close`` when you're done feeding in your 
           document if ``done`` is not already ``True``. 
        &quot;&quot;&quot;</span>
        <span class="s3">if </span><span class="s1">self.done:</span>
            <span class="s3">return</span>

        <span class="s3">if not </span><span class="s1">byte_str:</span>
            <span class="s3">return</span>

        <span class="s3">if not </span><span class="s1">isinstance(byte_str</span><span class="s3">, </span><span class="s1">bytearray):</span>
            <span class="s1">byte_str = bytearray(byte_str)</span>

        <span class="s0"># First check for known BOMs, since these are guaranteed to be correct</span>
        <span class="s3">if not </span><span class="s1">self._got_data:</span>
            <span class="s0"># If the data starts with BOM, we know it is UTF</span>
            <span class="s3">if </span><span class="s1">byte_str.startswith(codecs.BOM_UTF8):</span>
                <span class="s0"># EF BB BF  UTF-8 with BOM</span>
                <span class="s1">self.result = {</span>
                    <span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s6">&quot;UTF-8-SIG&quot;</span><span class="s3">,</span>
                    <span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">,</span>
                    <span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s3">,</span>
                <span class="s1">}</span>
            <span class="s3">elif </span><span class="s1">byte_str.startswith((codecs.BOM_UTF32_LE</span><span class="s3">, </span><span class="s1">codecs.BOM_UTF32_BE)):</span>
                <span class="s0"># FF FE 00 00  UTF-32, little-endian BOM</span>
                <span class="s0"># 00 00 FE FF  UTF-32, big-endian BOM</span>
                <span class="s1">self.result = {</span><span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s6">&quot;UTF-32&quot;</span><span class="s3">, </span><span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">, </span><span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s1">}</span>
            <span class="s3">elif </span><span class="s1">byte_str.startswith(</span><span class="s5">b&quot;</span><span class="s3">\xFE\xFF\x00\x00</span><span class="s5">&quot;</span><span class="s1">):</span>
                <span class="s0"># FE FF 00 00  UCS-4, unusual octet order BOM (3412)</span>
                <span class="s1">self.result = {</span>
                    <span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s6">&quot;X-ISO-10646-UCS-4-3412&quot;</span><span class="s3">,</span>
                    <span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">,</span>
                    <span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s3">,</span>
                <span class="s1">}</span>
            <span class="s3">elif </span><span class="s1">byte_str.startswith(</span><span class="s5">b&quot;</span><span class="s3">\x00\x00\xFF\xFE</span><span class="s5">&quot;</span><span class="s1">):</span>
                <span class="s0"># 00 00 FF FE  UCS-4, unusual octet order BOM (2143)</span>
                <span class="s1">self.result = {</span>
                    <span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s6">&quot;X-ISO-10646-UCS-4-2143&quot;</span><span class="s3">,</span>
                    <span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">,</span>
                    <span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s3">,</span>
                <span class="s1">}</span>
            <span class="s3">elif </span><span class="s1">byte_str.startswith((codecs.BOM_LE</span><span class="s3">, </span><span class="s1">codecs.BOM_BE)):</span>
                <span class="s0"># FF FE  UTF-16, little endian BOM</span>
                <span class="s0"># FE FF  UTF-16, big endian BOM</span>
                <span class="s1">self.result = {</span><span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s6">&quot;UTF-16&quot;</span><span class="s3">, </span><span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">, </span><span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s1">}</span>

            <span class="s1">self._got_data = </span><span class="s3">True</span>
            <span class="s3">if </span><span class="s1">self.result[</span><span class="s6">&quot;encoding&quot;</span><span class="s1">] </span><span class="s3">is not None</span><span class="s1">:</span>
                <span class="s1">self.done = </span><span class="s3">True</span>
                <span class="s3">return</span>

        <span class="s0"># If none of those matched and we've only see ASCII so far, check</span>
        <span class="s0"># for high bytes and escape sequences</span>
        <span class="s3">if </span><span class="s1">self._input_state == InputState.PURE_ASCII:</span>
            <span class="s3">if </span><span class="s1">self.HIGH_BYTE_DETECTOR.search(byte_str):</span>
                <span class="s1">self._input_state = InputState.HIGH_BYTE</span>
            <span class="s3">elif </span><span class="s1">(</span>
                <span class="s1">self._input_state == InputState.PURE_ASCII</span>
                <span class="s3">and </span><span class="s1">self.ESC_DETECTOR.search(self._last_char + byte_str)</span>
            <span class="s1">):</span>
                <span class="s1">self._input_state = InputState.ESC_ASCII</span>

        <span class="s1">self._last_char = byte_str[-</span><span class="s4">1</span><span class="s1">:]</span>

        <span class="s0"># next we will look to see if it is appears to be either a UTF-16 or</span>
        <span class="s0"># UTF-32 encoding</span>
        <span class="s3">if not </span><span class="s1">self._utf1632_prober:</span>
            <span class="s1">self._utf1632_prober = UTF1632Prober()</span>

        <span class="s3">if </span><span class="s1">self._utf1632_prober.state == ProbingState.DETECTING:</span>
            <span class="s3">if </span><span class="s1">self._utf1632_prober.feed(byte_str) == ProbingState.FOUND_IT:</span>
                <span class="s1">self.result = {</span>
                    <span class="s6">&quot;encoding&quot;</span><span class="s1">: self._utf1632_prober.charset_name</span><span class="s3">,</span>
                    <span class="s6">&quot;confidence&quot;</span><span class="s1">: self._utf1632_prober.get_confidence()</span><span class="s3">,</span>
                    <span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s3">,</span>
                <span class="s1">}</span>
                <span class="s1">self.done = </span><span class="s3">True</span>
                <span class="s3">return</span>

        <span class="s0"># If we've seen escape sequences, use the EscCharSetProber, which</span>
        <span class="s0"># uses a simple state machine to check for known escape sequences in</span>
        <span class="s0"># HZ and ISO-2022 encodings, since those are the only encodings that</span>
        <span class="s0"># use such sequences.</span>
        <span class="s3">if </span><span class="s1">self._input_state == InputState.ESC_ASCII:</span>
            <span class="s3">if not </span><span class="s1">self._esc_charset_prober:</span>
                <span class="s1">self._esc_charset_prober = EscCharSetProber(self.lang_filter)</span>
            <span class="s3">if </span><span class="s1">self._esc_charset_prober.feed(byte_str) == ProbingState.FOUND_IT:</span>
                <span class="s1">self.result = {</span>
                    <span class="s6">&quot;encoding&quot;</span><span class="s1">: self._esc_charset_prober.charset_name</span><span class="s3">,</span>
                    <span class="s6">&quot;confidence&quot;</span><span class="s1">: self._esc_charset_prober.get_confidence()</span><span class="s3">,</span>
                    <span class="s6">&quot;language&quot;</span><span class="s1">: self._esc_charset_prober.language</span><span class="s3">,</span>
                <span class="s1">}</span>
                <span class="s1">self.done = </span><span class="s3">True</span>
        <span class="s0"># If we've seen high bytes (i.e., those with values greater than 127),</span>
        <span class="s0"># we need to do more complicated checks using all our multi-byte and</span>
        <span class="s0"># single-byte probers that are left.  The single-byte probers</span>
        <span class="s0"># use character bigram distributions to determine the encoding, whereas</span>
        <span class="s0"># the multi-byte probers use a combination of character unigram and</span>
        <span class="s0"># bigram distributions.</span>
        <span class="s3">elif </span><span class="s1">self._input_state == InputState.HIGH_BYTE:</span>
            <span class="s3">if not </span><span class="s1">self._charset_probers:</span>
                <span class="s1">self._charset_probers = [MBCSGroupProber(self.lang_filter)]</span>
                <span class="s0"># If we're checking non-CJK encodings, use single-byte prober</span>
                <span class="s3">if </span><span class="s1">self.lang_filter &amp; LanguageFilter.NON_CJK:</span>
                    <span class="s1">self._charset_probers.append(SBCSGroupProber())</span>
                <span class="s1">self._charset_probers.append(Latin1Prober())</span>
            <span class="s3">for </span><span class="s1">prober </span><span class="s3">in </span><span class="s1">self._charset_probers:</span>
                <span class="s3">if </span><span class="s1">prober.feed(byte_str) == ProbingState.FOUND_IT:</span>
                    <span class="s1">self.result = {</span>
                        <span class="s6">&quot;encoding&quot;</span><span class="s1">: prober.charset_name</span><span class="s3">,</span>
                        <span class="s6">&quot;confidence&quot;</span><span class="s1">: prober.get_confidence()</span><span class="s3">,</span>
                        <span class="s6">&quot;language&quot;</span><span class="s1">: prober.language</span><span class="s3">,</span>
                    <span class="s1">}</span>
                    <span class="s1">self.done = </span><span class="s3">True</span>
                    <span class="s3">break</span>
            <span class="s3">if </span><span class="s1">self.WIN_BYTE_DETECTOR.search(byte_str):</span>
                <span class="s1">self._has_win_bytes = </span><span class="s3">True</span>

    <span class="s3">def </span><span class="s1">close(self):</span>
        <span class="s2">&quot;&quot;&quot; 
        Stop analyzing the current document and come up with a final 
        prediction. 
 
        :returns:  The ``result`` attribute, a ``dict`` with the keys 
                   `encoding`, `confidence`, and `language`. 
        &quot;&quot;&quot;</span>
        <span class="s0"># Don't bother with checks if we're already done</span>
        <span class="s3">if </span><span class="s1">self.done:</span>
            <span class="s3">return </span><span class="s1">self.result</span>
        <span class="s1">self.done = </span><span class="s3">True</span>

        <span class="s3">if not </span><span class="s1">self._got_data:</span>
            <span class="s1">self.logger.debug(</span><span class="s6">&quot;no data received!&quot;</span><span class="s1">)</span>

        <span class="s0"># Default to ASCII if it is all we've seen so far</span>
        <span class="s3">elif </span><span class="s1">self._input_state == InputState.PURE_ASCII:</span>
            <span class="s1">self.result = {</span><span class="s6">&quot;encoding&quot;</span><span class="s1">: </span><span class="s6">&quot;ascii&quot;</span><span class="s3">, </span><span class="s6">&quot;confidence&quot;</span><span class="s1">: </span><span class="s4">1.0</span><span class="s3">, </span><span class="s6">&quot;language&quot;</span><span class="s1">: </span><span class="s6">&quot;&quot;</span><span class="s1">}</span>

        <span class="s0"># If we have seen non-ASCII, return the best that met MINIMUM_THRESHOLD</span>
        <span class="s3">elif </span><span class="s1">self._input_state == InputState.HIGH_BYTE:</span>
            <span class="s1">prober_confidence = </span><span class="s3">None</span>
            <span class="s1">max_prober_confidence = </span><span class="s4">0.0</span>
