    <span class="s2">if not </span><span class="s1">asdict </span><span class="s2">or not </span><span class="s1">(project_table </span><span class="s2">or </span><span class="s1">setuptools_table):</span>

    <span class="s2">if </span><span class="s1">setuptools_table:</span>
        <span class="s3"># TODO: Remove the following once the feature stabilizes:</span>
        <span class="s1">msg = </span><span class="s4">&quot;Support for `[tool.setuptools]` in `pyproject.toml` is still *beta*.&quot;</span>

        <span class="s3"># by the _ignore_errors context manager.</span>
        <span class="s2">if </span><span class="s1">ignore_option_errors:</span>

        <span class="s2">return </span><span class="s1">expand_configuration(asdict</span><span class="s2">, </span><span class="s1">root_dir</span><span class="s2">, </span><span class="s1">ignore_option_errors</span><span class="s2">, </span><span class="s1">dist)</span>

<span class="s2">def </span><span class="s1">_skip_bad_config(</span>
    <span class="s1">project_cfg: dict</span><span class="s2">, </span><span class="s1">setuptools_cfg: dict</span><span class="s2">, </span><span class="s1">dist: Optional[</span><span class="s4">&quot;Distribution&quot;</span><span class="s1">]</span>
<span class="s1">) -&gt; bool:</span>
    <span class="s0">&quot;&quot;&quot;Be temporarily forgiving with invalid ``pyproject.toml``&quot;&quot;&quot;</span>
    <span class="s3"># See pypa/setuptools#3199 and pypa/cibuildwheel#1064</span>

    <span class="s2">if </span><span class="s1">dist </span><span class="s2">is None or </span><span class="s1">(</span>
        <span class="s1">dist.metadata.name </span><span class="s2">is None</span>
        <span class="s2">and </span><span class="s1">dist.metadata.version </span><span class="s2">is None</span>
        <span class="s2">and </span><span class="s1">dist.install_requires </span><span class="s2">is None</span>
    <span class="s1">):</span>
        <span class="s3"># It seems that the build is not getting any configuration from other places</span>
        <span class="s2">return False</span>

    <span class="s2">if </span><span class="s1">setuptools_cfg:</span>
        <span class="s3"># If `[tool.setuptools]` is set, then `pyproject.toml` config is intentional</span>
        <span class="s2">return False</span>

    <span class="s1">given_config = set(project_cfg.keys())</span>
    <span class="s1">popular_subset = {</span><span class="s4">&quot;name&quot;</span><span class="s2">, </span><span class="s4">&quot;version&quot;</span><span class="s2">, </span><span class="s4">&quot;python_requires&quot;</span><span class="s2">, </span><span class="s4">&quot;requires-python&quot;</span><span class="s1">}</span>
    <span class="s2">if </span><span class="s1">given_config &lt;= popular_subset:</span>
        <span class="s3"># It seems that the docs in cibuildtool has been inadvertently encouraging users</span>
        <span class="s3"># to create `pyproject.toml` files that are not compliant with the standards.</span>
        <span class="s3"># Let's be forgiving for the time being.</span>
        <span class="s1">warnings.warn(_InvalidFile.message()</span><span class="s2">, </span><span class="s1">_InvalidFile</span><span class="s2">, </span><span class="s1">stacklevel=</span><span class="s5">2</span><span class="s1">)</span>
        <span class="s2">return True</span>

    <span class="s2">return False</span>
