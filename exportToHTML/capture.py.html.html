            <span class="s1">open(os.dup(f.fileno())</span><span class="s2">, </span><span class="s1">mode</span><span class="s2">, </span><span class="s1">buffering)</span><span class="s2">,</span>

            <span class="s1">f.errors</span><span class="s2">,</span>
            <span class="s1">f.newlines</span><span class="s2">,</span>
            <span class="s1">f.line_buffering</span><span class="s2">,</span>
