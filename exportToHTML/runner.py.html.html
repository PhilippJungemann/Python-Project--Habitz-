<span class="s2">def </span><span class="s1">pytest_terminal_summary(terminalreporter: </span><span class="s4">&quot;TerminalReporter&quot;</span><span class="s1">) -&gt; </span><span class="s2">None</span><span class="s1">:</span>

                <span class="s1">% (len(dlist) - i</span><span class="s2">, </span><span class="s1">durations_min)</span>
            <span class="s1">)</span>
            <span class="s2">break</span>
        <span class="s1">tr.write_line(</span><span class="s4">f&quot;</span><span class="s2">{</span><span class="s1">rep.duration</span><span class="s2">:</span><span class="s4">02.2f</span><span class="s2">}</span><span class="s4">s </span><span class="s2">{</span><span class="s1">rep.when</span><span class="s2">:</span><span class="s4">&lt;8</span><span class="s2">} {</span><span class="s1">rep.nodeid</span><span class="s2">}</span><span class="s4">&quot;</span><span class="s1">)</span>

<span class="s2">def </span><span class="s1">runtestprotocol(</span>

<span class="s2">class </span><span class="s1">SetupState:</span>
    <span class="s0">&quot;&quot;&quot;Shared state for setting up/tearing down test items or collectors 
    in a session. 
 
    Suppose we have a collection tree as follows: 
 
    &lt;Session session&gt; 
        &lt;Module mod1&gt; 
            &lt;Function item1&gt; 
        &lt;Module mod2&gt; 
            &lt;Function item2&gt; 
 
    The SetupState maintains a stack. The stack starts out empty: 
 
        [] 
 
    During the setup phase of item1, setup(item1) is called. What it does 
    is: 
 
        push session to stack, run session.setup() 
        push mod1 to stack, run mod1.setup() 
        push item1 to stack, run item1.setup() 
 
    The stack is: 
 
        [session, mod1, item1] 
 
    While the stack is in this shape, it is allowed to add finalizers to 
    each of session, mod1, item1 using addfinalizer(). 
 
    During the teardown phase of item1, teardown_exact(item2) is called, 
    where item2 is the next item to item1. What it does is: 
 
        pop item1 from stack, run its teardowns 
        pop mod1 from stack, run its teardowns 
 
    mod1 was popped because it ended its purpose with item1. The stack is: 
 
        [session] 
 
    During the setup phase of item2, setup(item2) is called. What it does 
    is: 
 
        push mod2 to stack, run mod2.setup() 
        push item2 to stack, run item2.setup() 
 
    Stack: 
 
        [session, mod2, item2] 
 
    During the teardown phase of item2, teardown_exact(None) is called, 
    because item2 is the last item. What it does is: 
 
        pop item2 from stack, run its teardowns 
        pop mod2 from stack, run its teardowns 
        pop session from stack, run its teardowns 
 
    Stack: 
 
        [] 
 
    The end! 
    &quot;&quot;&quot;</span>

    <span class="s2">def </span><span class="s1">__init__(self) -&gt; </span><span class="s2">None</span><span class="s1">:</span>
        <span class="s3"># The stack is in the dict insertion order.</span>
        <span class="s1">self.stack: Dict[</span>
            <span class="s1">Node</span><span class="s2">,</span>
            <span class="s1">Tuple[</span>
                <span class="s3"># Node's finalizers.</span>
                <span class="s1">List[Callable[[]</span><span class="s2">, </span><span class="s1">object]]</span><span class="s2">,</span>
                <span class="s3"># Node's exception, if its setup raised.</span>
                <span class="s1">Optional[Union[OutcomeException</span><span class="s2">, </span><span class="s1">Exception]]</span><span class="s2">,</span>
            <span class="s1">]</span><span class="s2">,</span>
        <span class="s1">] = {}</span>

    <span class="s2">def </span><span class="s1">setup(self</span><span class="s2">, </span><span class="s1">item: Item) -&gt; </span><span class="s2">None</span><span class="s1">:</span>
        <span class="s0">&quot;&quot;&quot;Setup objects along the collector chain to the item.&quot;&quot;&quot;</span>
        <span class="s1">needed_collectors = item.listchain()</span>

        <span class="s3"># If a collector fails its setup, fail its entire subtree of items.</span>
        <span class="s3"># The setup is not retried for each item - the same exception is used.</span>
        <span class="s2">for </span><span class="s1">col</span><span class="s2">, </span><span class="s1">(finalizers</span><span class="s2">, </span><span class="s1">exc) </span><span class="s2">in </span><span class="s1">self.stack.items():</span>
            <span class="s2">assert </span><span class="s1">col </span><span class="s2">in </span><span class="s1">needed_collectors</span><span class="s2">, </span><span class="s4">&quot;previous item was not torn down properly&quot;</span>
            <span class="s2">if </span><span class="s1">exc:</span>
                <span class="s2">raise </span><span class="s1">exc</span>

        <span class="s2">for </span><span class="s1">col </span><span class="s2">in </span><span class="s1">needed_collectors[len(self.stack) :]:</span>
            <span class="s2">assert </span><span class="s1">col </span><span class="s2">not in </span><span class="s1">self.stack</span>
            <span class="s3"># Push onto the stack.</span>
            <span class="s1">self.stack[col] = ([col.teardown]</span><span class="s2">, None</span><span class="s1">)</span>
            <span class="s2">try</span><span class="s1">:</span>
                <span class="s1">col.setup()</span>
            <span class="s2">except </span><span class="s1">TEST_OUTCOME </span><span class="s2">as </span><span class="s1">exc:</span>
                <span class="s1">self.stack[col] = (self.stack[col][</span><span class="s5">0</span><span class="s1">]</span><span class="s2">, </span><span class="s1">exc)</span>
                <span class="s2">raise </span><span class="s1">exc</span>

    <span class="s2">def </span><span class="s1">addfinalizer(self</span><span class="s2">, </span><span class="s1">finalizer: Callable[[]</span><span class="s2">, </span><span class="s1">object]</span><span class="s2">, </span><span class="s1">node: Node) -&gt; </span><span class="s2">None</span><span class="s1">:</span>
        <span class="s0">&quot;&quot;&quot;Attach a finalizer to the given node. 
 
        The node must be currently active in the stack. 
        &quot;&quot;&quot;</span>
        <span class="s2">assert </span><span class="s1">node </span><span class="s2">and not </span><span class="s1">isinstance(node</span><span class="s2">, </span><span class="s1">tuple)</span>
        <span class="s2">assert </span><span class="s1">callable(finalizer)</span>
        <span class="s2">assert </span><span class="s1">node </span><span class="s2">in </span><span class="s1">self.stack</span><span class="s2">, </span><span class="s1">(node</span><span class="s2">, </span><span class="s1">self.stack)</span>
        <span class="s1">self.stack[node][</span><span class="s5">0</span><span class="s1">].append(finalizer)</span>

    <span class="s2">def </span><span class="s1">teardown_exact(self</span><span class="s2">, </span><span class="s1">nextitem: Optional[Item]) -&gt; </span><span class="s2">None</span><span class="s1">:</span>
        <span class="s0">&quot;&quot;&quot;Teardown the current stack up until reaching nodes that nextitem

        When nextitem is None (meaning we're at the last item), the entire 
        stack is torn down. 
        &quot;&quot;&quot;</span>
        <span class="s1">needed_collectors = nextitem </span><span class="s2">and </span><span class="s1">nextitem.listchain() </span><span class="s2">or </span><span class="s1">[]</span>
        <span class="s1">exc = </span><span class="s2">None</span>
        <span class="s2">while </span><span class="s1">self.stack:</span>
            <span class="s2">if </span><span class="s1">list(self.stack.keys()) == needed_collectors[: len(self.stack)]:</span>
                <span class="s2">break</span>
            <span class="s1">node</span><span class="s2">, </span><span class="s1">(finalizers</span><span class="s2">, </span><span class="s1">_) = self.stack.popitem()</span>
            <span class="s2">while </span><span class="s1">finalizers:</span>
                <span class="s1">fin = finalizers.pop()</span>
