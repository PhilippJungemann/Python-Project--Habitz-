<span class="s2">class </span><span class="s1">CCompiler:</span>
    <span class="s0">&quot;&quot;&quot;Abstract base class to define the interface that must be implemented 
    by real compiler classes.  Also has some utility methods used by 
    several compiler classes. 
 
    The basic idea behind a compiler abstraction class is that each 
    instance can be used for all the compile/link steps in building a 
    single project.  Thus, attributes common to all of those compile and 
    link steps -- include directories, macros to define, libraries to link 
    against, etc. -- are attributes of the compiler instance.  To allow for 
    variability in how individual files are treated, most of those 
    attributes may be varied on a per-compilation or per-link basis. 
    &quot;&quot;&quot;</span>

    <span class="s3"># 'isinstance'.  In concrete CCompiler subclasses, 'compiler_type'</span>
    <span class="s3"># should really, really be one of the keys of the 'compiler_class'</span>
    <span class="s3"># dictionary (see below -- used by the 'new_compiler()' factory</span>
    <span class="s3"># function) -- authors of new compiler interface classes are</span>
    <span class="s3"># responsible for updating 'compiler_class'!</span>

    <span class="s3">#   * client can't provide additional options for a compiler,</span>
    <span class="s3">#     e.g. warning, optimization, debugging flags.  Perhaps this</span>
    <span class="s3">#     should be the domain of concrete compiler abstraction classes</span>
    <span class="s3">#     (UnixCCompiler, MSVCCompiler, etc.) -- or perhaps the base</span>
    <span class="s3">#     class should have methods for the common ones.</span>
    <span class="s3">#   * can't completely override the include or library searchg</span>

    <span class="s3">#     sure how useful it is; maybe for cross-compiling, but</span>

    <span class="s3">#     compilers probably have a dedicated binary with the</span>

    <span class="s3">#   * can't do really freaky things with the library list/library</span>

    <span class="s3">#     different versions of libfoo.a in different locations.  I</span>

        <span class="s0">&quot;&quot;&quot;Define the executables (and options for them) that will be run 
        to perform the various stages of compilation.  The exact set of 
        executables that may be specified here depends on the compiler

        &quot;&quot;&quot;</span>

        <span class="s3"># basically the same things with Unix C compilers.</span>

    <span class="s3"># The default implementation of the filename generating methods are</span>
    <span class="s3"># prejudiced towards the Unix/DOS/Windows view of the world:</span>
    <span class="s3">#   * object files are named by replacing the source file extension</span>
    <span class="s3">#     (eg. .c/.cpp -&gt; .o/.obj)</span>
    <span class="s3">#   * library files (shared or static) are named by plugging the</span>
    <span class="s3">#     library name and extension into a format string, eg.</span>
    <span class="s3">#     &quot;lib%s.%s&quot; % (lib_name, &quot;.a&quot;) for Unix static libraries</span>
    <span class="s3">#   * executables are named by appending an extension (possibly</span>
    <span class="s3">#     empty) to the program name: eg. progname + &quot;.exe&quot; for</span>
    <span class="s3">#     Windows</span>
    <span class="s3">#</span>
    <span class="s3"># To reduce redundant code, these methods expect to find</span>
    <span class="s3"># several attributes in the current object (presumably defined</span>
    <span class="s3"># as class attributes):</span>
    <span class="s3">#   * src_extensions -</span>
    <span class="s3">#     list of C/C++ source file extensions, eg. ['.c', '.cpp']</span>
    <span class="s3">#   * obj_extension -</span>
    <span class="s3">#     object file extension, eg. '.o' or '.obj'</span>
    <span class="s3">#   * static_lib_extension -</span>
    <span class="s3">#     extension for static library files, eg. '.a' or '.lib'</span>
    <span class="s3">#   * shared_lib_extension -</span>
    <span class="s3">#     extension for shared library/object files, eg. '.so', '.dll'</span>
    <span class="s3">#   * static_lib_format -</span>
    <span class="s3">#     format string for generating static library filenames,</span>
    <span class="s3">#     eg. 'lib%s.%s' or '%s.%s'</span>
    <span class="s3">#   * shared_lib_format</span>
    <span class="s3">#     format string for generating shared library filenames</span>
    <span class="s3">#     (probably same as static_lib_format, since the extension</span>
    <span class="s3">#     is one of the intended parameters to the format string)</span>
    <span class="s3">#   * exe_extension -</span>
    <span class="s3">#     extension for executable files, eg. '' or '.exe'</span>

            <span class="s2">return </span><span class="s1">compiler</span>
    <span class="s3"># Default to Unix compiler</span>
    <span class="s2">return </span><span class="s4">'unix'</span>

<span class="s2">def </span><span class="s1">new_compiler(plat=</span><span class="s2">None, </span><span class="s1">compiler=</span><span class="s2">None, </span><span class="s1">verbose=</span><span class="s5">0</span><span class="s2">, </span><span class="s1">dry_run=</span><span class="s5">0</span><span class="s2">, </span><span class="s1">force=</span><span class="s5">0</span><span class="s1">):</span>
    <span class="s0">&quot;&quot;&quot;Generate an instance of some CCompiler subclass for the supplied 
    platform/compiler combination.  'plat' defaults to 'os.name' 
    (eg. 'posix', 'nt'), and 'compiler' defaults to the default compiler 
    for that platform.  Currently only 'posix' and 'nt' are supported, and 
    the default compilers are &quot;traditional Unix interface&quot; (UnixCCompiler 
    class) and Visual C++ (MSVCCompiler class).  Note that it's perfectly 
    possible to ask for a Unix compiler object under Windows, and a 
    Microsoft compiler object under Unix -- if you supply a value for 
    'compiler', 'plat' is ignored. 
    &quot;&quot;&quot;</span>
    <span class="s2">if </span><span class="s1">plat </span><span class="s2">is None</span><span class="s1">:</span>
        <span class="s1">plat = os.name</span>

    <span class="s2">try</span><span class="s1">:</span>
        <span class="s2">if </span><span class="s1">compiler </span><span class="s2">is None</span><span class="s1">:</span>
            <span class="s1">compiler = get_default_compiler(plat)</span>

        <span class="s1">(module_name</span><span class="s2">, </span><span class="s1">class_name</span><span class="s2">, </span><span class="s1">long_description) = compiler_class[compiler]</span>
    <span class="s2">except </span><span class="s1">KeyError:</span>
        <span class="s1">msg = </span><span class="s4">&quot;don't know how to compile C/C++ code on platform '%s'&quot; </span><span class="s1">% plat</span>
        <span class="s2">if </span><span class="s1">compiler </span><span class="s2">is not None</span><span class="s1">:</span>
            <span class="s1">msg = msg + </span><span class="s4">&quot; with '%s' compiler&quot; </span><span class="s1">% compiler</span>
        <span class="s2">raise </span><span class="s1">DistutilsPlatformError(msg)</span>

    <span class="s2">try</span><span class="s1">:</span>
        <span class="s1">module_name = </span><span class="s4">&quot;distutils.&quot; </span><span class="s1">+ module_name</span>
        <span class="s1">__import__(module_name)</span>
        <span class="s1">module = sys.modules[module_name]</span>
        <span class="s1">klass = vars(module)[class_name]</span>
    <span class="s2">except </span><span class="s1">ImportError:</span>
        <span class="s2">raise </span><span class="s1">DistutilsModuleError(</span>
            <span class="s4">&quot;can't compile C/C++ code: unable to load module '%s'&quot; </span><span class="s1">% module_name</span>
        <span class="s1">)</span>
    <span class="s2">except </span><span class="s1">KeyError:</span>
        <span class="s2">raise </span><span class="s1">DistutilsModuleError(</span>
            <span class="s4">&quot;can't compile C/C++ code: unable to find class '%s' &quot;</span>
            <span class="s4">&quot;in module '%s'&quot; </span><span class="s1">% (class_name</span><span class="s2">, </span><span class="s1">module_name)</span>
        <span class="s1">)</span>

    <span class="s3"># XXX The None is necessary to preserve backwards compatibility</span>
    <span class="s3"># with classes that expect verbose to be the first positional</span>
    <span class="s3"># argument.</span>
    <span class="s2">return </span><span class="s1">klass(</span><span class="s2">None, </span><span class="s1">dry_run</span><span class="s2">, </span><span class="s1">force)</span>


<span class="s2">def </span><span class="s1">gen_preprocess_options(macros</span><span class="s2">, </span><span class="s1">include_dirs):</span>
    <span class="s0">&quot;&quot;&quot;Generate C pre-processor options (-D, -U, -I) as used by at least 
    two types of compilers: the typical Unix compiler and Visual C++. 
    'macros' is the usual thing, a list of 1- or 2-tuples, where (name,) 
    means undefine (-U) macro 'name', and (name,value) means define (-D) 
    macro 'name' to 'value'.  'include_dirs' is just a list of directory 
    names to be added to the header file search path (-I).  Returns a list 
    of command-line options suitable for either Unix compilers or Visual 
    C++. 
    &quot;&quot;&quot;</span>
    <span class="s3"># XXX it would be nice (mainly aesthetic, and so we don't generate</span>
    <span class="s3"># stupid-looking command lines) to go over 'macros' and eliminate</span>
    <span class="s3"># redundant definitions/undefinitions (ie. ensure that only the</span>
    <span class="s3"># latest mention of a particular macro winds up on the command</span>
    <span class="s3"># line).  I don't think it's essential, though, since most (all?)</span>
    <span class="s3"># Unix C compilers only pay attention to the latest -D or -U</span>
    <span class="s3"># mention of a macro on their command line.  Similar situation for</span>
    <span class="s3"># 'include_dirs'.  I'm punting on both for now.  Anyways, weeding out</span>
    <span class="s3"># redundancies like this should probably be the province of</span>
    <span class="s3"># CCompiler, since the data structures used are inherited from it</span>
    <span class="s3"># and therefore common to all CCompiler classes.</span>
    <span class="s1">pp_opts = []</span>
    <span class="s2">for </span><span class="s1">macro </span><span class="s2">in </span><span class="s1">macros:</span>
        <span class="s2">if not </span><span class="s1">(isinstance(macro</span><span class="s2">, </span><span class="s1">tuple) </span><span class="s2">and </span><span class="s5">1 </span><span class="s1">&lt;= len(macro) &lt;= </span><span class="s5">2</span><span class="s1">):</span>
            <span class="s2">raise </span><span class="s1">TypeError(</span>
                <span class="s4">&quot;bad macro definition '%s': &quot;</span>
                <span class="s4">&quot;each element of 'macros' list must be a 1- or 2-tuple&quot; </span><span class="s1">% macro</span>
            <span class="s1">)</span>

        <span class="s2">if </span><span class="s1">len(macro) == </span><span class="s5">1</span><span class="s1">:  </span><span class="s3"># undefine this macro</span>
            <span class="s1">pp_opts.append(</span><span class="s4">&quot;-U%s&quot; </span><span class="s1">% macro[</span><span class="s5">0</span><span class="s1">])</span>
        <span class="s2">elif </span><span class="s1">len(macro) == </span><span class="s5">2</span><span class="s1">:</span>
            <span class="s2">if </span><span class="s1">macro[</span><span class="s5">1</span><span class="s1">] </span><span class="s2">is None</span><span class="s1">:  </span><span class="s3"># define with no explicit value</span>
                <span class="s1">pp_opts.append(</span><span class="s4">&quot;-D%s&quot; </span><span class="s1">% macro[</span><span class="s5">0</span><span class="s1">])</span>
            <span class="s2">else</span><span class="s1">:</span>
                <span class="s3"># XXX *don't* need to be clever about quoting the</span>
                <span class="s3"># macro value here, because we're going to avoid the</span>
                <span class="s3"># shell at all costs when we spawn the command!</span>
                <span class="s1">pp_opts.append(</span><span class="s4">&quot;-D%s=%s&quot; </span><span class="s1">% macro)</span>

    <span class="s2">for </span><span class="s1">dir </span><span class="s2">in </span><span class="s1">include_dirs:</span>
        <span class="s1">pp_opts.append(</span><span class="s4">&quot;-I%s&quot; </span><span class="s1">% dir)</span>
    <span class="s2">return </span><span class="s1">pp_opts</span>


<span class="s2">def </span><span class="s1">gen_lib_options(compiler</span><span class="s2">, </span><span class="s1">library_dirs</span><span class="s2">, </span><span class="s1">runtime_library_dirs</span><span class="s2">, </span><span class="s1">libraries):</span>
    <span class="s0">&quot;&quot;&quot;Generate linker options for searching library directories and 
    linking with specific libraries.  'libraries' and 'library_dirs' are, 
    respectively, lists of library names (not filenames!) and search 
    directories.  Returns a list of command-line options suitable for use 
    with some compiler (depending on the two format strings passed in). 
    &quot;&quot;&quot;</span>
    <span class="s1">lib_opts = []</span>

    <span class="s2">for </span><span class="s1">dir </span><span class="s2">in </span><span class="s1">library_dirs:</span>
        <span class="s1">lib_opts.append(compiler.library_dir_option(dir))</span>

    <span class="s2">for </span><span class="s1">dir </span><span class="s2">in </span><span class="s1">runtime_library_dirs:</span>
        <span class="s1">opt = compiler.runtime_library_dir_option(dir)</span>
        <span class="s2">if </span><span class="s1">isinstance(opt</span><span class="s2">, </span><span class="s1">list):</span>
            <span class="s1">lib_opts = lib_opts + opt</span>
        <span class="s2">else</span><span class="s1">:</span>
            <span class="s1">lib_opts.append(opt)</span>
