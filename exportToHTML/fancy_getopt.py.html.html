<pre><span class="s0">&quot;&quot;&quot;distutils.fancy_getopt 
 
Wrapper around the standard getopt module that provides the following 
additional features: 
  * short and long options are tied together 
  * options have help strings, so fancy_getopt could potentially 
    create a complete usage summary 
  * options set attributes of a passed-in object 
&quot;&quot;&quot;</span>

<span class="s2">import </span><span class="s1">sys</span><span class="s2">, </span><span class="s1">string</span><span class="s2">, </span><span class="s1">re</span>
<span class="s2">import </span><span class="s1">getopt</span>
<span class="s2">from </span><span class="s1">distutils.errors </span><span class="s2">import </span><span class="s1">*</span>

<span class="s3"># Much like command_re in distutils.core, this is close to but not quite</span>
<span class="s3"># the same as a Python NAME -- except, in the spirit of most GNU</span>
<span class="s3"># utilities, we use '-' in place of '_'.  (The spirit of LISP lives on!)</span>
<span class="s3"># The similarities to NAME are again not a coincidence...</span>
<span class="s1">longopt_pat = </span><span class="s4">r'[a-zA-Z](?:[a-zA-Z0-9-]*)'</span>
<span class="s1">longopt_re = re.compile(</span><span class="s4">r'^%s$' </span><span class="s1">% longopt_pat)</span>

<span class="s3"># For recognizing &quot;negative alias&quot; options, eg. &quot;quiet=!verbose&quot;</span>
<span class="s1">neg_alias_re = re.compile(</span><span class="s4">&quot;^(%s)=!(%s)$&quot; </span><span class="s1">% (longopt_pat</span><span class="s2">, </span><span class="s1">longopt_pat))</span>

<span class="s3"># This is used to translate long options to legitimate Python identifiers</span>
<span class="s3"># (for use as attributes of some object).</span>
<span class="s1">longopt_xlate = str.maketrans(</span><span class="s4">'-'</span><span class="s2">, </span><span class="s4">'_'</span><span class="s1">)</span>


<span class="s2">class </span><span class="s1">FancyGetopt:</span>
    <span class="s0">&quot;&quot;&quot;Wrapper around the standard 'getopt()' module that provides some 
    handy extra functionality: 
      * short and long options are tied together 
      * options have help strings, and help text can be assembled 
        from them 
      * options set attributes of a passed-in object 
      * boolean options can have &quot;negative aliases&quot; -- eg. if 
        --quiet is the &quot;negative alias&quot; of --verbose, then &quot;--quiet&quot; 
        on the command line sets 'verbose' to false 
    &quot;&quot;&quot;</span>

    <span class="s2">def </span><span class="s1">__init__(self</span><span class="s2">, </span><span class="s1">option_table=</span><span class="s2">None</span><span class="s1">):</span>
        <span class="s3"># The option table is (currently) a list of tuples.  The</span>
        <span class="s3"># tuples may have 3 or four values:</span>
        <span class="s3">#   (long_option, short_option, help_string [, repeatable])</span>
        <span class="s3"># if an option takes an argument, its long_option should have '='</span>
        <span class="s3"># appended; short_option should just be a single character, no ':'</span>
        <span class="s3"># in any case.  If a long_option doesn't have a corresponding</span>
        <span class="s3"># short_option, short_option should be None.  All option tuples</span>
        <span class="s3"># must have long options.</span>
        <span class="s1">self.option_table = option_table</span>

        <span class="s3"># 'option_index' maps long option names to entries in the option</span>
        <span class="s3"># table (ie. those 3-tuples).</span>
        <span class="s1">self.option_index = {}</span>
        <span class="s2">if </span><span class="s1">self.option_table:</span>
            <span class="s1">self._build_index()</span>

        <span class="s3"># 'alias' records (duh) alias options; {'foo': 'bar'} means</span>
        <span class="s3"># --foo is an alias for --bar</span>
        <span class="s1">self.alias = {}</span>

        <span class="s3"># 'negative_alias' keeps track of options that are the boolean</span>
        <span class="s3"># opposite of some other option</span>
        <span class="s1">self.negative_alias = {}</span>

        <span class="s3"># These keep track of the information in the option table.  We</span>
        <span class="s3"># don't actually populate these structures until we're ready to</span>
        <span class="s3"># parse the command-line, since the 'option_table' passed in here</span>
        <span class="s3"># isn't necessarily the final word.</span>
        <span class="s1">self.short_opts = []</span>
        <span class="s1">self.long_opts = []</span>
        <span class="s1">self.short2long = {}</span>
        <span class="s1">self.attr_name = {}</span>
        <span class="s1">self.takes_arg = {}</span>

        <span class="s3"># And 'option_order' is filled up in 'getopt()'; it records the</span>
        <span class="s3"># original order of options (and their values) on the command-line,</span>
        <span class="s3"># but expands short options, converts aliases, etc.</span>
        <span class="s1">self.option_order = []</span>

    <span class="s2">def </span><span class="s1">_build_index(self):</span>
        <span class="s1">self.option_index.clear()</span>
        <span class="s2">for </span><span class="s1">option </span><span class="s2">in </span><span class="s1">self.option_table:</span>
            <span class="s1">self.option_index[option[</span><span class="s5">0</span><span class="s1">]] = option</span>

    <span class="s2">def </span><span class="s1">set_option_table(self</span><span class="s2">, </span><span class="s1">option_table):</span>
        <span class="s1">self.option_table = option_table</span>
        <span class="s1">self._build_index()</span>

    <span class="s2">def </span><span class="s1">add_option(self</span><span class="s2">, </span><span class="s1">long_option</span><span class="s2">, </span><span class="s1">short_option=</span><span class="s2">None, </span><span class="s1">help_string=</span><span class="s2">None</span><span class="s1">):</span>
        <span class="s2">if </span><span class="s1">long_option </span><span class="s2">in </span><span class="s1">self.option_index:</span>
            <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                <span class="s4">&quot;option conflict: already an option '%s'&quot; </span><span class="s1">% long_option</span>
            <span class="s1">)</span>
        <span class="s2">else</span><span class="s1">:</span>
            <span class="s1">option = (long_option</span><span class="s2">, </span><span class="s1">short_option</span><span class="s2">, </span><span class="s1">help_string)</span>
            <span class="s1">self.option_table.append(option)</span>
            <span class="s1">self.option_index[long_option] = option</span>

    <span class="s2">def </span><span class="s1">has_option(self</span><span class="s2">, </span><span class="s1">long_option):</span>
        <span class="s0">&quot;&quot;&quot;Return true if the option table for this parser has an 
        option with long name 'long_option'.&quot;&quot;&quot;</span>
        <span class="s2">return </span><span class="s1">long_option </span><span class="s2">in </span><span class="s1">self.option_index</span>

    <span class="s2">def </span><span class="s1">get_attr_name(self</span><span class="s2">, </span><span class="s1">long_option):</span>
        <span class="s0">&quot;&quot;&quot;Translate long option name 'long_option' to the form it 
        has as an attribute of some object: ie., translate hyphens 
        to underscores.&quot;&quot;&quot;</span>
        <span class="s2">return </span><span class="s1">long_option.translate(longopt_xlate)</span>

    <span class="s2">def </span><span class="s1">_check_alias_dict(self</span><span class="s2">, </span><span class="s1">aliases</span><span class="s2">, </span><span class="s1">what):</span>
        <span class="s2">assert </span><span class="s1">isinstance(aliases</span><span class="s2">, </span><span class="s1">dict)</span>
        <span class="s2">for </span><span class="s1">(alias</span><span class="s2">, </span><span class="s1">opt) </span><span class="s2">in </span><span class="s1">aliases.items():</span>
            <span class="s2">if </span><span class="s1">alias </span><span class="s2">not in </span><span class="s1">self.option_index:</span>
                <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                    <span class="s1">(</span><span class="s4">&quot;invalid %s '%s': &quot; &quot;option '%s' not defined&quot;</span><span class="s1">)</span>
                    <span class="s1">% (what</span><span class="s2">, </span><span class="s1">alias</span><span class="s2">, </span><span class="s1">alias)</span>
                <span class="s1">)</span>
            <span class="s2">if </span><span class="s1">opt </span><span class="s2">not in </span><span class="s1">self.option_index:</span>
                <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                    <span class="s1">(</span><span class="s4">&quot;invalid %s '%s': &quot; &quot;aliased option '%s' not defined&quot;</span><span class="s1">)</span>
                    <span class="s1">% (what</span><span class="s2">, </span><span class="s1">alias</span><span class="s2">, </span><span class="s1">opt)</span>
                <span class="s1">)</span>

    <span class="s2">def </span><span class="s1">set_aliases(self</span><span class="s2">, </span><span class="s1">alias):</span>
        <span class="s0">&quot;&quot;&quot;Set the aliases for this option parser.&quot;&quot;&quot;</span>
        <span class="s1">self._check_alias_dict(alias</span><span class="s2">, </span><span class="s4">&quot;alias&quot;</span><span class="s1">)</span>
        <span class="s1">self.alias = alias</span>

    <span class="s2">def </span><span class="s1">set_negative_aliases(self</span><span class="s2">, </span><span class="s1">negative_alias):</span>
        <span class="s0">&quot;&quot;&quot;Set the negative aliases for this option parser. 
        'negative_alias' should be a dictionary mapping option names to 
        option names, both the key and value must already be defined 
        in the option table.&quot;&quot;&quot;</span>
        <span class="s1">self._check_alias_dict(negative_alias</span><span class="s2">, </span><span class="s4">&quot;negative alias&quot;</span><span class="s1">)</span>
        <span class="s1">self.negative_alias = negative_alias</span>

    <span class="s2">def </span><span class="s1">_grok_option_table(self):</span>
        <span class="s0">&quot;&quot;&quot;Populate the various data structures that keep tabs on the 
        option table.  Called by 'getopt()' before it can do anything 
        worthwhile. 
        &quot;&quot;&quot;</span>
        <span class="s1">self.long_opts = []</span>
        <span class="s1">self.short_opts = []</span>
        <span class="s1">self.short2long.clear()</span>
        <span class="s1">self.repeat = {}</span>

        <span class="s2">for </span><span class="s1">option </span><span class="s2">in </span><span class="s1">self.option_table:</span>
            <span class="s2">if </span><span class="s1">len(option) == </span><span class="s5">3</span><span class="s1">:</span>
                <span class="s1">long</span><span class="s2">, </span><span class="s1">short</span><span class="s2">, </span><span class="s1">help = option</span>
                <span class="s1">repeat = </span><span class="s5">0</span>
            <span class="s2">elif </span><span class="s1">len(option) == </span><span class="s5">4</span><span class="s1">:</span>
                <span class="s1">long</span><span class="s2">, </span><span class="s1">short</span><span class="s2">, </span><span class="s1">help</span><span class="s2">, </span><span class="s1">repeat = option</span>
            <span class="s2">else</span><span class="s1">:</span>
                <span class="s3"># the option table is part of the code, so simply</span>
                <span class="s3"># assert that it is correct</span>
                <span class="s2">raise </span><span class="s1">ValueError(</span><span class="s4">&quot;invalid option tuple: %r&quot; </span><span class="s1">% (option</span><span class="s2">,</span><span class="s1">))</span>

            <span class="s3"># Type- and value-check the option names</span>
            <span class="s2">if not </span><span class="s1">isinstance(long</span><span class="s2">, </span><span class="s1">str) </span><span class="s2">or </span><span class="s1">len(long) &lt; </span><span class="s5">2</span><span class="s1">:</span>
                <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                    <span class="s1">(</span><span class="s4">&quot;invalid long option '%s': &quot; &quot;must be a string of length &gt;= 2&quot;</span><span class="s1">)</span>
                    <span class="s1">% long</span>
                <span class="s1">)</span>

            <span class="s2">if not </span><span class="s1">((short </span><span class="s2">is None</span><span class="s1">) </span><span class="s2">or </span><span class="s1">(isinstance(short</span><span class="s2">, </span><span class="s1">str) </span><span class="s2">and </span><span class="s1">len(short) == </span><span class="s5">1</span><span class="s1">)):</span>
                <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                    <span class="s4">&quot;invalid short option '%s': &quot;</span>
                    <span class="s4">&quot;must a single character or None&quot; </span><span class="s1">% short</span>
                <span class="s1">)</span>

            <span class="s1">self.repeat[long] = repeat</span>
            <span class="s1">self.long_opts.append(long)</span>

            <span class="s2">if </span><span class="s1">long[-</span><span class="s5">1</span><span class="s1">] == </span><span class="s4">'='</span><span class="s1">:  </span><span class="s3"># option takes an argument?</span>
                <span class="s2">if </span><span class="s1">short:</span>
                    <span class="s1">short = short + </span><span class="s4">':'</span>
                <span class="s1">long = long[</span><span class="s5">0</span><span class="s1">:-</span><span class="s5">1</span><span class="s1">]</span>
                <span class="s1">self.takes_arg[long] = </span><span class="s5">1</span>
            <span class="s2">else</span><span class="s1">:</span>
                <span class="s3"># Is option is a &quot;negative alias&quot; for some other option (eg.</span>
                <span class="s3"># &quot;quiet&quot; == &quot;!verbose&quot;)?</span>
                <span class="s1">alias_to = self.negative_alias.get(long)</span>
                <span class="s2">if </span><span class="s1">alias_to </span><span class="s2">is not None</span><span class="s1">:</span>
                    <span class="s2">if </span><span class="s1">self.takes_arg[alias_to]:</span>
                        <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                            <span class="s4">&quot;invalid negative alias '%s': &quot;</span>
                            <span class="s4">&quot;aliased option '%s' takes a value&quot; </span><span class="s1">% (long</span><span class="s2">, </span><span class="s1">alias_to)</span>
                        <span class="s1">)</span>

                    <span class="s1">self.long_opts[-</span><span class="s5">1</span><span class="s1">] = long  </span><span class="s3"># XXX redundant?!</span>
                <span class="s1">self.takes_arg[long] = </span><span class="s5">0</span>

            <span class="s3"># If this is an alias option, make sure its &quot;takes arg&quot; flag is</span>
            <span class="s3"># the same as the option it's aliased to.</span>
            <span class="s1">alias_to = self.alias.get(long)</span>
            <span class="s2">if </span><span class="s1">alias_to </span><span class="s2">is not None</span><span class="s1">:</span>
                <span class="s2">if </span><span class="s1">self.takes_arg[long] != self.takes_arg[alias_to]:</span>
                    <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                        <span class="s4">&quot;invalid alias '%s': inconsistent with &quot;</span>
                        <span class="s4">&quot;aliased option '%s' (one of them takes a value, &quot;</span>
                        <span class="s4">&quot;the other doesn't&quot; </span><span class="s1">% (long</span><span class="s2">, </span><span class="s1">alias_to)</span>
                    <span class="s1">)</span>

            <span class="s3"># Now enforce some bondage on the long option name, so we can</span>
            <span class="s3"># later translate it to an attribute name on some object.  Have</span>
            <span class="s3"># to do this a bit late to make sure we've removed any trailing</span>
            <span class="s3"># '='.</span>
            <span class="s2">if not </span><span class="s1">longopt_re.match(long):</span>
                <span class="s2">raise </span><span class="s1">DistutilsGetoptError(</span>
                    <span class="s4">&quot;invalid long option name '%s' &quot;</span>
                    <span class="s4">&quot;(must be letters, numbers, hyphens only&quot; </span><span class="s1">% long</span>
                <span class="s1">)</span>

            <span class="s1">self.attr_name[long] = self.get_attr_name(long)</span>
            <span class="s2">if </span><span class="s1">short:</span>
                <span class="s1">self.short_opts.append(short)</span>
                <span class="s1">self.short2long[short[</span><span class="s5">0</span><span class="s1">]] = long</span>

    <span class="s2">def </span><span class="s1">getopt(self</span><span class="s2">, </span><span class="s1">args=</span><span class="s2">None, </span><span class="s1">object=</span><span class="s2">None</span><span class="s1">):</span>

            <span class="s2">if </span><span class="s1">alias:</span>

            <span class="s2">if not </span><span class="s1">self.takes_arg[opt]:  </span><span class="s3"># boolean option?</span>

            <span class="s1">attr = self.attr_name[opt]</span>
            <span class="s3"># The only repeating option at the moment is 'verbose'.</span>

            <span class="s2">if </span><span class="s1">val </span><span class="s2">and </span><span class="s1">self.repeat.get(attr) </span><span class="s2">is not None</span><span class="s1">:</span>

            <span class="s1">self.option_order.append((opt</span><span class="s2">, </span><span class="s1">val))</span>

        <span class="s2">if </span><span class="s1">created_object:</span>
