<span class="s2">def </span><span class="s1">enquote_executable(executable):</span>
    <span class="s2">if </span><span class="s3">' ' </span><span class="s2">in </span><span class="s1">executable:</span>
        <span class="s0"># make sure we quote only the executable in case of env</span>
        <span class="s0"># for example /usr/bin/env &quot;/dir with spaces/bin/jython&quot;</span>
        <span class="s0"># instead of &quot;/usr/bin/env /dir with spaces/bin/jython&quot;</span>
        <span class="s0"># otherwise whole</span>
        <span class="s2">if </span><span class="s1">executable.startswith(</span><span class="s3">'/usr/bin/env '</span><span class="s1">):</span>
            <span class="s1">env</span><span class="s2">, </span><span class="s1">_executable = executable.split(</span><span class="s3">' '</span><span class="s2">, </span><span class="s5">1</span><span class="s1">)</span>

    <span class="s2">def </span><span class="s1">_build_shebang(self</span><span class="s2">, </span><span class="s1">executable</span><span class="s2">, </span><span class="s1">post_interp):</span>
        <span class="s6">&quot;&quot;&quot; 
        Build a shebang line. In the simple case (on Windows, or a shebang line 
        which is not too long or contains spaces) use a simple formulation for 
        the shebang. Otherwise, use /bin/sh as the executable, with a contrived 
        shebang which allows the script to run either under Python or sh, using 
        suitable quoting. Thanks to Harald Nordgren for his input. 
 
        See also: http://www.in-ulm.de/~mascheck/various/shebang/#length 
                  https://hg.mozilla.org/mozilla-central/file/tip/mach 
        &quot;&quot;&quot;</span>
        <span class="s2">if </span><span class="s1">os.name != </span><span class="s3">'posix'</span><span class="s1">:</span>
            <span class="s1">simple_shebang = </span><span class="s2">True</span>
        <span class="s2">else</span><span class="s1">:</span>
            <span class="s0"># Add 3 for '#!' prefix and newline suffix.</span>
            <span class="s1">shebang_length = len(executable) + len(post_interp) + </span><span class="s5">3</span>
            <span class="s2">if </span><span class="s1">sys.platform == </span><span class="s3">'darwin'</span><span class="s1">:</span>
                <span class="s1">max_shebang_length = </span><span class="s5">512</span>
            <span class="s2">else</span><span class="s1">:</span>
                <span class="s1">max_shebang_length = </span><span class="s5">127</span>
            <span class="s1">simple_shebang = ((</span><span class="s4">b' ' </span><span class="s2">not in </span><span class="s1">executable) </span><span class="s2">and</span>
                              <span class="s1">(shebang_length &lt;= max_shebang_length))</span>

        <span class="s2">if </span><span class="s1">simple_shebang:</span>
            <span class="s1">result = </span><span class="s4">b'#!' </span><span class="s1">+ executable + post_interp + </span><span class="s4">b'</span><span class="s2">\n</span><span class="s4">'</span>
        <span class="s2">else</span><span class="s1">:</span>
            <span class="s1">result = </span><span class="s4">b'#!/bin/sh</span><span class="s2">\n</span><span class="s4">'</span>
            <span class="s1">result += </span><span class="s4">b&quot;'''exec' &quot; </span><span class="s1">+ executable + post_interp + </span><span class="s4">b' &quot;$0&quot; &quot;$@&quot;</span><span class="s2">\n</span><span class="s4">'</span>
            <span class="s1">result += </span><span class="s4">b&quot;' '''&quot;</span>

        <span class="s2">elif </span><span class="s1">in_venv():  </span><span class="s0"># pragma: no cover</span>

                <span class="s1">sysconfig.get_config_var(</span><span class="s3">'BINDIR'</span><span class="s1">)</span><span class="s2">,</span>
               <span class="s3">'python%s%s' </span><span class="s1">% (sysconfig.get_config_var(</span><span class="s3">'VERSION'</span><span class="s1">)</span><span class="s2">,</span>

                <span class="s0"># a version suffix are created, so we use python.exe</span>

                                <span class="s3">'python%s' </span><span class="s1">% (sysconfig.get_config_var(</span><span class="s3">'EXE'</span><span class="s1">)))</span>

        <span class="s2">if </span><span class="s1">sys.platform.startswith(</span><span class="s3">'java'</span><span class="s1">):  </span><span class="s0"># pragma: no cover</span>

        <span class="s0"># Normalise case for Windows - COMMENTED OUT</span>
        <span class="s0"># executable = os.path.normcase(executable)</span>
        <span class="s0"># N.B. The normalising operation above has been commented out: See</span>
        <span class="s0"># issue #124. Although paths in Windows are generally case-insensitive,</span>

        <span class="s0"># Python parser starts to read a script using UTF-8 until</span>
        <span class="s0"># it gets a #coding:xxx cookie. The shebang has to be the</span>
        <span class="s0"># first line of a file, the #coding:xxx cookie cannot be</span>
        <span class="s0"># written before. So the shebang has to be decodable from</span>
        <span class="s0"># UTF-8.</span>
        <span class="s2">try</span><span class="s1">:</span>
            <span class="s1">shebang.decode(</span><span class="s3">'utf-8'</span><span class="s1">)</span>
        <span class="s2">except </span><span class="s1">UnicodeDecodeError:  </span><span class="s0"># pragma: no cover</span>
            <span class="s2">raise </span><span class="s1">ValueError(</span>
                <span class="s3">'The shebang (%r) is not decodable from utf-8' </span><span class="s1">% shebang)</span>
        <span class="s0"># If the script is encoded to a custom encoding (use a</span>
        <span class="s0"># #coding:xxx cookie), the shebang has to be decodable from</span>
        <span class="s0"># the script encoding too.</span>
        <span class="s2">if </span><span class="s1">encoding != </span><span class="s3">'utf-8'</span><span class="s1">:</span>
            <span class="s2">try</span><span class="s1">:</span>
                <span class="s1">shebang.decode(encoding)</span>
            <span class="s2">except </span><span class="s1">UnicodeDecodeError:  </span><span class="s0"># pragma: no cover</span>
                <span class="s2">raise </span><span class="s1">ValueError(</span>
                    <span class="s3">'The shebang (%r) is not decodable '</span>
                    <span class="s3">'from the script encoding (%r)' </span><span class="s1">% (shebang</span><span class="s2">, </span><span class="s1">encoding))</span>
