<pre><span class="s0">&quot;&quot;&quot;Automatic discovery of Python modules and packages (for inclusion in the 
distribution) and other config values. 
 
For the purposes of this module, the following nomenclature is used: 
 
- &quot;src-layout&quot;: a directory representing a Python project that contains a &quot;src&quot; 
  folder. Everything under the &quot;src&quot; folder is meant to be included in the 
  distribution when packaging the project. Example:: 
 
    . 
    ├── tox.ini 
    ├── pyproject.toml 
    └── src/ 
        └── mypkg/ 
            ├── __init__.py 
            ├── mymodule.py 
            └── my_data_file.txt 
 
- &quot;flat-layout&quot;: a Python project that does not use &quot;src-layout&quot; but instead 
  have a directory under the project root for each package:: 
 
    . 
    ├── tox.ini 
    ├── pyproject.toml 
    └── mypkg/ 
        ├── __init__.py 
        ├── mymodule.py 
        └── my_data_file.txt 
 
- &quot;single-module&quot;: a project that contains a single Python script direct under 
  the project root (no directory used):: 
 
    . 
    ├── tox.ini 
    ├── pyproject.toml 
    └── mymodule.py 
 
&quot;&quot;&quot;</span>

<span class="s2">class </span><span class="s1">_Finder:</span>
    <span class="s0">&quot;&quot;&quot;Base class that exposes functionality for module/package finders&quot;&quot;&quot;</span>

    <span class="s1">ALWAYS_EXCLUDE: Tuple[str</span><span class="s2">, </span><span class="s1">...] = ()</span>
    <span class="s1">DEFAULT_EXCLUDE: Tuple[str</span><span class="s2">, </span><span class="s1">...] = ()</span>

    <span class="s1">@classmethod</span>
    <span class="s2">def </span><span class="s1">find(</span>
        <span class="s1">cls</span><span class="s2">,</span>
        <span class="s1">where: _Path = </span><span class="s4">'.'</span><span class="s2">,</span>
        <span class="s1">exclude: Iterable[str] = ()</span><span class="s2">,</span>
        <span class="s1">include: Iterable[str] = (</span><span class="s4">'*'</span><span class="s2">,</span><span class="s1">)</span>
    <span class="s1">) -&gt; List[str]:</span>
        <span class="s0">&quot;&quot;&quot;Return a list of all Python items (packages or modules, depending on 
        the finder implementation) found within directory 'where'. 
 
        'where' is the root directory which will be searched. 
        It should be supplied as a &quot;cross-platform&quot; (i.e. URL-style) path; 
        it will be converted to the appropriate local path syntax. 
 
        'exclude' is a sequence of names to exclude; '*' can be used 
        as a wildcard in the names. 
        When finding packages, 'foo.*' will exclude all subpackages of 'foo' 
        (but not 'foo' itself). 
 
        'include' is a sequence of names to include. 
        If it's specified, only the named items will be included. 
        If it's not specified, all found items will be included. 
        'include' can contain shell style wildcard patterns just like 
        'exclude'. 
        &quot;&quot;&quot;</span>

        <span class="s1">exclude = exclude </span><span class="s2">or </span><span class="s1">cls.DEFAULT_EXCLUDE</span>
        <span class="s2">return </span><span class="s1">list(</span>
            <span class="s1">cls._find_iter(</span>
                <span class="s1">convert_path(str(where))</span><span class="s2">,</span>
                <span class="s1">cls._build_filter(*cls.ALWAYS_EXCLUDE</span><span class="s2">, </span><span class="s1">*exclude)</span><span class="s2">,</span>
                <span class="s1">cls._build_filter(*include)</span><span class="s2">,</span>
            <span class="s1">)</span>
        <span class="s1">)</span>

    <span class="s1">@classmethod</span>
    <span class="s2">def </span><span class="s1">_find_iter(cls</span><span class="s2">, </span><span class="s1">where: _Path</span><span class="s2">, </span><span class="s1">exclude: _Filter</span><span class="s2">, </span><span class="s1">include: _Filter) -&gt; StrIter:</span>
        <span class="s2">raise </span><span class="s1">NotImplementedError</span>

    <span class="s1">@staticmethod</span>
    <span class="s2">def </span><span class="s1">_build_filter(*patterns: str) -&gt; _Filter:</span>
        <span class="s0">&quot;&quot;&quot; 
        Given a list of patterns, return a callable that will be true only if 
        the input matches at least one of the patterns. 
        &quot;&quot;&quot;</span>
        <span class="s2">return lambda </span><span class="s1">name: any(fnmatchcase(name</span><span class="s2">, </span><span class="s1">pat) </span><span class="s2">for </span><span class="s1">pat </span><span class="s2">in </span><span class="s1">patterns)</span>


<span class="s2">class </span><span class="s1">PackageFinder(_Finder):</span>
    <span class="s0">&quot;&quot;&quot; 
    Generate a list of all Python packages found within a directory 
    &quot;&quot;&quot;</span>

    <span class="s1">ALWAYS_EXCLUDE = (</span><span class="s4">&quot;ez_setup&quot;</span><span class="s2">, </span><span class="s4">&quot;*__pycache__&quot;</span><span class="s1">)</span>

    <span class="s1">@classmethod</span>
    <span class="s2">def </span><span class="s1">_find_iter(cls</span><span class="s2">, </span><span class="s1">where: _Path</span><span class="s2">, </span><span class="s1">exclude: _Filter</span><span class="s2">, </span><span class="s1">include: _Filter) -&gt; StrIter:</span>
        <span class="s0">&quot;&quot;&quot; 
        All the packages found in 'where' that pass the 'include' filter, but 
        not the 'exclude' filter. 
        &quot;&quot;&quot;</span>
        <span class="s2">for </span><span class="s1">root</span><span class="s2">, </span><span class="s1">dirs</span><span class="s2">, </span><span class="s1">files </span><span class="s2">in </span><span class="s1">os.walk(str(where)</span><span class="s2">, </span><span class="s1">followlinks=</span><span class="s2">True</span><span class="s1">):</span>
            <span class="s3"># Copy dirs to iterate over it, then empty dirs.</span>
            <span class="s1">all_dirs = dirs[:]</span>
            <span class="s1">dirs[:] = []</span>

            <span class="s2">for </span><span class="s1">dir </span><span class="s2">in </span><span class="s1">all_dirs:</span>
                <span class="s1">full_path = os.path.join(root</span><span class="s2">, </span><span class="s1">dir)</span>
                <span class="s1">rel_path = os.path.relpath(full_path</span><span class="s2">, </span><span class="s1">where)</span>
                <span class="s1">package = rel_path.replace(os.path.sep</span><span class="s2">, </span><span class="s4">'.'</span><span class="s1">)</span>

                <span class="s3"># Skip directory trees that are not valid packages</span>
                <span class="s2">if </span><span class="s4">'.' </span><span class="s2">in </span><span class="s1">dir </span><span class="s2">or not </span><span class="s1">cls._looks_like_package(full_path</span><span class="s2">, </span><span class="s1">package):</span>
                    <span class="s2">continue</span>

                <span class="s3"># Should this package be included?</span>
                <span class="s2">if </span><span class="s1">include(package) </span><span class="s2">and not </span><span class="s1">exclude(package):</span>
                    <span class="s2">yield </span><span class="s1">package</span>

                <span class="s3"># Keep searching subdirectories, as there may be more packages</span>
                <span class="s3"># down there, even if the parent was excluded.</span>
                <span class="s1">dirs.append(dir)</span>

    <span class="s1">@staticmethod</span>
    <span class="s2">def </span><span class="s1">_looks_like_package(path: _Path</span><span class="s2">, </span><span class="s1">_package_name: str) -&gt; bool:</span>
        <span class="s0">&quot;&quot;&quot;Does a directory look like a package?&quot;&quot;&quot;</span>
        <span class="s2">return </span><span class="s1">os.path.isfile(os.path.join(path</span><span class="s2">, </span><span class="s4">'__init__.py'</span><span class="s1">))</span>

<span class="s2">class </span><span class="s1">ModuleFinder(_Finder):</span>
    <span class="s0">&quot;&quot;&quot;Find isolated Python modules. 
    This function will **not** recurse subdirectories. 
    &quot;&quot;&quot;</span>

    <span class="s1">@classmethod</span>
    <span class="s2">def </span><span class="s1">_find_iter(cls</span><span class="s2">, </span><span class="s1">where: _Path</span><span class="s2">, </span><span class="s1">exclude: _Filter</span><span class="s2">, </span><span class="s1">include: _Filter) -&gt; StrIter:</span>

<span class="s2">def </span><span class="s1">remove_nested_packages(packages: List[str]) -&gt; List[str]:</span>

<span class="s2">def </span><span class="s1">find_parent_package(</span>
    <span class="s1">packages: List[str]</span><span class="s2">, </span><span class="s1">package_dir: Dict[str</span><span class="s2">, </span><span class="s1">str]</span><span class="s2">, </span><span class="s1">root_dir: _Path</span>

    <span class="s0">&quot;&quot;&quot;Find the parent package that is not a namespace.&quot;&quot;&quot;</span>

    <span class="s1">common_ancestors = []</span>
    <span class="s2">for </span><span class="s1">i</span><span class="s2">, </span><span class="s1">name </span><span class="s2">in </span><span class="s1">enumerate(packages):</span>

            <span class="s3"># the list will be empty</span>
            <span class="s2">break</span>
