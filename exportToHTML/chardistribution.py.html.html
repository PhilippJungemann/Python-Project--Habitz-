<span class="s2">class </span><span class="s1">CharDistributionAnalysis:</span>
    <span class="s1">ENOUGH_DATA_THRESHOLD = </span><span class="s3">1024</span>
    <span class="s1">SURE_YES = </span><span class="s3">0.99</span>
    <span class="s1">SURE_NO = </span><span class="s3">0.01</span>
    <span class="s1">MINIMUM_DATA_THRESHOLD = </span><span class="s3">3</span>

    <span class="s2">def </span><span class="s1">__init__(self):</span>
        <span class="s0"># Mapping table to get frequency order from char order (get from</span>
        <span class="s0"># GetOrder())</span>
        <span class="s1">self._char_to_freq_order = tuple()</span>
        <span class="s1">self._table_size = </span><span class="s2">None  </span><span class="s0"># Size of above table</span>
        <span class="s0"># This is a constant value which varies from language to language,</span>
        <span class="s0"># used in calculating confidence.  See</span>
        <span class="s0"># http://www.mozilla.org/projects/intl/UniversalCharsetDetection.html</span>
        <span class="s0"># for further detail.</span>
        <span class="s1">self.typical_distribution_ratio = </span><span class="s2">None</span>
        <span class="s1">self._done = </span><span class="s2">None</span>
        <span class="s1">self._total_chars = </span><span class="s2">None</span>
        <span class="s1">self._freq_chars = </span><span class="s2">None</span>
        <span class="s1">self.reset()</span>

    <span class="s2">def </span><span class="s1">reset(self):</span>
        <span class="s4">&quot;&quot;&quot;reset analyser, clear any state&quot;&quot;&quot;</span>
        <span class="s0"># If this flag is set to True, detection is done and conclusion has</span>
        <span class="s0"># been made</span>
        <span class="s1">self._done = </span><span class="s2">False</span>
        <span class="s1">self._total_chars = </span><span class="s3">0  </span><span class="s0"># Total characters encountered</span>
        <span class="s0"># The number of characters whose frequency order is less than 512</span>
        <span class="s1">self._freq_chars = </span><span class="s3">0</span>

    <span class="s2">def </span><span class="s1">feed(self</span><span class="s2">, </span><span class="s1">char</span><span class="s2">, </span><span class="s1">char_len):</span>
        <span class="s4">&quot;&quot;&quot;feed a character with known length&quot;&quot;&quot;</span>
        <span class="s2">if </span><span class="s1">char_len == </span><span class="s3">2</span><span class="s1">:</span>
            <span class="s0"># we only care about 2-bytes character in our distribution analysis</span>
            <span class="s1">order = self.get_order(char)</span>
        <span class="s2">else</span><span class="s1">:</span>
            <span class="s1">order = -</span><span class="s3">1</span>
        <span class="s2">if </span><span class="s1">order &gt;= </span><span class="s3">0</span><span class="s1">:</span>
            <span class="s1">self._total_chars += </span><span class="s3">1</span>
            <span class="s0"># order is valid</span>
            <span class="s2">if </span><span class="s1">order &lt; self._table_size:</span>
                <span class="s2">if </span><span class="s3">512 </span><span class="s1">&gt; self._char_to_freq_order[order]:</span>
                    <span class="s1">self._freq_chars += </span><span class="s3">1</span>

    <span class="s2">def </span><span class="s1">get_confidence(self):</span>
        <span class="s4">&quot;&quot;&quot;return confidence based on existing data&quot;&quot;&quot;</span>
        <span class="s0"># if we didn't receive any character in our consideration range,</span>
        <span class="s0"># return negative answer</span>
        <span class="s2">if </span><span class="s1">self._total_chars &lt;= </span><span class="s3">0 </span><span class="s2">or </span><span class="s1">self._freq_chars &lt;= self.MINIMUM_DATA_THRESHOLD:</span>
            <span class="s2">return </span><span class="s1">self.SURE_NO</span>

        <span class="s2">if </span><span class="s1">self._total_chars != self._freq_chars:</span>
            <span class="s1">r = self._freq_chars / (</span>
                <span class="s1">(self._total_chars - self._freq_chars) * self.typical_distribution_ratio</span>
            <span class="s1">)</span>
            <span class="s2">if </span><span class="s1">r &lt; self.SURE_YES:</span>
                <span class="s2">return </span><span class="s1">r</span>

        <span class="s0"># normalize confidence (we don't want to be 100% sure)</span>
        <span class="s2">return </span><span class="s1">self.SURE_YES</span>

    <span class="s2">def </span><span class="s1">got_enough_data(self):</span>
        <span class="s0"># It is not necessary to receive all data to draw conclusion.</span>
        <span class="s0"># For charset detection, certain amount of data is enough</span>
        <span class="s2">return </span><span class="s1">self._total_chars &gt; self.ENOUGH_DATA_THRESHOLD</span>

    <span class="s2">def </span><span class="s1">get_order(self</span><span class="s2">, </span><span class="s1">_):</span>
        <span class="s0"># We do not handle characters based on the original encoding string,</span>
        <span class="s0"># but convert this encoding string to a number, here called order.</span>
        <span class="s0"># This allows multiple encodings of a language to share one frequency</span>
        <span class="s0"># table.</span>
        <span class="s2">return </span><span class="s1">-</span><span class="s3">1</span>
