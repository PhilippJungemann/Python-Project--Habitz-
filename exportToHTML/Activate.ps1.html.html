<pre><span class="s0">&lt;#</span>
<span class="s1">.</span><span class="s2">Synopsis</span>
<span class="s0">Activate a Python virtual environment for the current PowerShell session. 
</span>
<span class="s1">.</span><span class="s2">Description</span>
<span class="s0">Pushes the python executable for a virtual environment to the front of the 
$Env:PATH environment variable and sets the prompt to signify that you are 
in a Python virtual environment. Makes use of the command line switches as 
well as the `pyvenv.cfg` file values present in the virtual environment. 
</span>
<span class="s1">.</span><span class="s2">Parameter VenvDir</span>
<span class="s0">Path to the directory that contains the virtual environment to activate. The 
default value for this is the parent of the directory that the Activate.ps1 
script is located within. 
</span>
<span class="s1">.</span><span class="s2">Parameter Prompt</span>
<span class="s0">The prompt prefix to display when this virtual environment is activated. By 
default, this prompt is the name of the virtual environment folder (VenvDir) 
surrounded by parentheses and followed by a single space (ie. '(.venv) '). 
</span>
<span class="s1">.</span><span class="s2">Example</span>
<span class="s0">Activate.ps1 
Activates the Python virtual environment that contains the Activate.ps1 script. 
</span>
<span class="s1">.</span><span class="s2">Example</span>
<span class="s0">Activate.ps1 -Verbose 
Activates the Python virtual environment that contains the Activate.ps1 script, 
and shows extra information about the activation as it executes. 
</span>
<span class="s1">.</span><span class="s2">Example</span>
<span class="s0">Activate.ps1 -VenvDir C:\Users\MyUser\Common\.venv 
Activates the Python virtual environment located in the specified location. 
</span>
<span class="s1">.</span><span class="s2">Example</span>
<span class="s0">Activate.ps1 -Prompt &quot;MyPython&quot; 
Activates the Python virtual environment that contains the Activate.ps1 script, 
and prefixes the current prompt with the specified string (surrounded in 
parentheses) while the virtual environment is active. 
</span>
<span class="s1">.</span><span class="s2">Notes</span>
<span class="s0">On Windows, it may be required to enable this Activate.ps1 script by setting the 
execution policy for the user. You can do this by issuing the following PowerShell 
command: 
</span>
<span class="s0">PS C:\&gt; Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser 
</span>
<span class="s0">For more information on Execution Policies:  
https://go.microsoft.com/fwlink/?LinkID=135170 
</span>
<span class="s0">#&gt;</span>
<span class="s5">Param</span><span class="s4">(</span>
    <span class="s4">[Parameter(</span><span class="s6">Mandatory </span><span class="s5">= </span><span class="s1">$false</span><span class="s4">)]</span>

<span class="s0">&lt;#</span>
<span class="s1">.</span><span class="s2">Description</span>
<span class="s0">Get-PyVenvConfig parses the values from the pyvenv.cfg file located in the 
given folder, and returns them in a map. 
</span>
<span class="s0">For each line in the pyvenv.cfg file, if that line can be parsed into exactly 
two strings separated by `=` (with any amount of whitespace surrounding the =) 
then it is considered a `key = value` line. The left hand string is the key, 
the right hand is the value. 
</span>
<span class="s0">If the value starts with a `'` or a `&quot;` then the first and last character is 
stripped from the value before being captured. 
</span>
<span class="s1">.</span><span class="s2">Parameter ConfigDir</span>
<span class="s0">Path to the directory that contains the `pyvenv.cfg` file. 
#&gt;</span>
<span class="s5">function </span><span class="s7">Get-PyVenvConfig</span><span class="s4">(</span>
    <span class="s4">[</span><span class="s5">String</span><span class="s4">]</span>
    <span class="s6">$ConfigDir</span>
<span class="s4">) {</span>
    <span class="s4">Write-Verbose </span><span class="s8">&quot;Given ConfigDir=</span><span class="s6">$ConfigDir</span><span class="s8">, obtain values in pyvenv.cfg&quot;</span>

    <span class="s0"># Ensure the file exists, and issue a warning if it doesn't (but still allow the function to continue). 
    </span><span class="s6">$pyvenvConfigPath </span><span class="s5">= </span><span class="s4">Join-Path </span><span class="s5">-</span><span class="s4">Resolve </span><span class="s5">-</span><span class="s4">Path </span><span class="s6">$ConfigDir </span><span class="s5">-</span><span class="s4">ChildPath </span><span class="s8">'pyvenv.cfg' </span><span class="s5">-</span><span class="s4">ErrorAction </span><span class="s5">Continue</span>

    <span class="s0"># An empty map will be returned if no config file is found. 
    </span><span class="s6">$pyvenvConfig </span><span class="s5">= @</span><span class="s4">{ }</span>

    <span class="s5">if </span><span class="s4">(</span><span class="s6">$pyvenvConfigPath</span><span class="s4">) {</span>

        <span class="s4">Write-Verbose </span><span class="s8">&quot;File exists, parse `key = value` lines&quot;</span>
        <span class="s6">$pyvenvConfigContent </span><span class="s5">= </span><span class="s4">Get-Content </span><span class="s5">-</span><span class="s4">Path </span><span class="s6">$pyvenvConfigPath</span>

        <span class="s6">$pyvenvConfigContent </span><span class="s5">| </span><span class="s4">ForEach-Object {</span>
            <span class="s6">$keyval </span><span class="s5">= </span><span class="s4">$PSItem </span><span class="s5">-split </span><span class="s8">&quot;\s*=\s*&quot;</span><span class="s5">, </span><span class="s9">2</span>
            <span class="s5">if </span><span class="s4">(</span><span class="s6">$keyval</span><span class="s4">[</span><span class="s9">0</span><span class="s4">] </span><span class="s5">-and </span><span class="s6">$keyval</span><span class="s4">[</span><span class="s9">1</span><span class="s4">]) {</span>
                <span class="s6">$val </span><span class="s5">= </span><span class="s6">$keyval</span><span class="s4">[</span><span class="s9">1</span><span class="s4">]</span>

                <span class="s0"># Remove extraneous quotations around a string value. 
                </span><span class="s5">if </span><span class="s4">(</span><span class="s8">&quot;'&quot;&quot;&quot;</span><span class="s4">.Contains(</span><span class="s6">$val.Substring</span><span class="s4">(</span><span class="s9">0</span><span class="s5">, </span><span class="s9">1</span><span class="s4">))) {</span>
                    <span class="s6">$val </span><span class="s5">= </span><span class="s6">$val.Substring</span><span class="s4">(</span><span class="s9">1</span><span class="s5">, </span><span class="s6">$val.Length </span><span class="s5">- </span><span class="s9">2</span><span class="s4">)</span>
                <span class="s4">}</span>

                <span class="s6">$pyvenvConfig</span><span class="s4">[</span><span class="s6">$keyval</span><span class="s4">[</span><span class="s9">0</span><span class="s4">]] </span><span class="s5">= </span><span class="s6">$val</span>
                <span class="s4">Write-Verbose </span><span class="s8">&quot;Adding Key: '</span><span class="s5">$</span><span class="s8">(</span><span class="s6">$keyval</span><span class="s8">[</span><span class="s9">0</span><span class="s8">])'='</span><span class="s6">$val</span><span class="s8">'&quot;</span>
            <span class="s4">}</span>
        <span class="s4">}</span>
    <span class="s4">}</span>
    <span class="s5">return </span><span class="s6">$pyvenvConfig</span>
<span class="s4">}</span>
