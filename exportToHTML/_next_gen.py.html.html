<span class="s3">def </span><span class="s1">define(</span>
    <span class="s1">maybe_cls=</span><span class="s3">None,</span>
    <span class="s1">*</span><span class="s3">,</span>
    <span class="s1">these=</span><span class="s3">None,</span>
    <span class="s1">repr=</span><span class="s3">None,</span>
    <span class="s1">hash=</span><span class="s3">None,</span>
    <span class="s1">init=</span><span class="s3">None,</span>
    <span class="s1">slots=</span><span class="s3">True,</span>
    <span class="s1">frozen=</span><span class="s3">False,</span>
    <span class="s1">weakref_slot=</span><span class="s3">True,</span>
    <span class="s1">str=</span><span class="s3">False,</span>
    <span class="s1">auto_attribs=</span><span class="s3">None,</span>
    <span class="s1">kw_only=</span><span class="s3">False,</span>
    <span class="s1">cache_hash=</span><span class="s3">False,</span>
    <span class="s1">auto_exc=</span><span class="s3">True,</span>
    <span class="s1">eq=</span><span class="s3">None,</span>
    <span class="s1">order=</span><span class="s3">False,</span>
    <span class="s1">auto_detect=</span><span class="s3">True,</span>
    <span class="s1">getstate_setstate=</span><span class="s3">None,</span>
    <span class="s1">on_setattr=</span><span class="s3">None,</span>
    <span class="s1">field_transformer=</span><span class="s3">None,</span>
    <span class="s1">match_args=</span><span class="s3">True,</span>
<span class="s1">):</span>
    <span class="s2">r&quot;&quot;&quot; 
    Define an ``attrs`` class. 
 
    Differences to the classic `attr.s` that it uses underneath: 
 
    - Automatically detect whether or not *auto_attribs* should be `True` (c.f. 
      *auto_attribs* parameter). 
    - If *frozen* is `False`, run converters and validators when setting an 
      attribute by default. 
    - *slots=True* 
 
      .. caution:: 
 
         Usually this has only upsides and few visible effects in everyday 
         programming. But it *can* lead to some suprising behaviors, so please 
         make sure to read :term:`slotted classes`. 
    - *auto_exc=True* 
    - *auto_detect=True* 
    - *order=False* 
    - Some options that were only relevant on Python 2 or were kept around for 
      backwards-compatibility have been removed. 
 
    Please note that these are all defaults and you can change them as you 
    wish. 
 
    :param Optional[bool] auto_attribs: If set to `True` or `False`, it behaves 
       exactly like `attr.s`. If left `None`, `attr.s` will try to guess: 
 
       1. If any attributes are annotated and no unannotated `attrs.fields`\ s 
          are found, it assumes *auto_attribs=True*. 
       2. Otherwise it assumes *auto_attribs=False* and tries to collect 
          `attrs.fields`\ s. 
 
    For now, please refer to `attr.s` for the rest of the parameters. 
 
    .. versionadded:: 20.1.0 
    .. versionchanged:: 21.3.0 Converters are also run ``on_setattr``. 
    &quot;&quot;&quot;</span>

    <span class="s3">def </span><span class="s1">do_it(cls</span><span class="s3">, </span><span class="s1">auto_attribs):</span>
        <span class="s3">return </span><span class="s1">attrs(</span>
            <span class="s1">maybe_cls=cls</span><span class="s3">,</span>
            <span class="s1">these=these</span><span class="s3">,</span>
            <span class="s1">repr=repr</span><span class="s3">,</span>
            <span class="s1">hash=hash</span><span class="s3">,</span>
            <span class="s1">init=init</span><span class="s3">,</span>
            <span class="s1">slots=slots</span><span class="s3">,</span>
            <span class="s1">frozen=frozen</span><span class="s3">,</span>
            <span class="s1">weakref_slot=weakref_slot</span><span class="s3">,</span>
            <span class="s1">str=str</span><span class="s3">,</span>
            <span class="s1">auto_attribs=auto_attribs</span><span class="s3">,</span>
            <span class="s1">kw_only=kw_only</span><span class="s3">,</span>
            <span class="s1">cache_hash=cache_hash</span><span class="s3">,</span>
            <span class="s1">auto_exc=auto_exc</span><span class="s3">,</span>
            <span class="s1">eq=eq</span><span class="s3">,</span>
            <span class="s1">order=order</span><span class="s3">,</span>
            <span class="s1">auto_detect=auto_detect</span><span class="s3">,</span>
            <span class="s1">collect_by_mro=</span><span class="s3">True,</span>
            <span class="s1">getstate_setstate=getstate_setstate</span><span class="s3">,</span>
            <span class="s1">on_setattr=on_setattr</span><span class="s3">,</span>
            <span class="s1">field_transformer=field_transformer</span><span class="s3">,</span>
            <span class="s1">match_args=match_args</span><span class="s3">,</span>
        <span class="s1">)</span>

    <span class="s3">def </span><span class="s1">wrap(cls):</span>
        <span class="s2">&quot;&quot;&quot; 
        Making this a wrapper ensures this code runs during class creation. 
 
        We also ensure that frozen-ness of classes is inherited. 
        &quot;&quot;&quot;</span>
        <span class="s3">nonlocal </span><span class="s1">frozen</span><span class="s3">, </span><span class="s1">on_setattr</span>

        <span class="s1">had_on_setattr = on_setattr </span><span class="s3">not in </span><span class="s1">(</span><span class="s3">None, </span><span class="s1">setters.NO_OP)</span>

        <span class="s0"># By default, mutable classes convert &amp; validate on setattr.</span>
        <span class="s3">if </span><span class="s1">frozen </span><span class="s3">is False and </span><span class="s1">on_setattr </span><span class="s3">is None</span><span class="s1">:</span>
            <span class="s1">on_setattr = _ng_default_on_setattr</span>

        <span class="s0"># However, if we subclass a frozen class, we inherit the immutability</span>
        <span class="s0"># and disable on_setattr.</span>
        <span class="s3">for </span><span class="s1">base_cls </span><span class="s3">in </span><span class="s1">cls.__bases__:</span>
            <span class="s3">if </span><span class="s1">base_cls.__setattr__ </span><span class="s3">is </span><span class="s1">_frozen_setattrs:</span>
                <span class="s3">if </span><span class="s1">had_on_setattr:</span>
                    <span class="s3">raise </span><span class="s1">ValueError(</span>
                        <span class="s4">&quot;Frozen classes can't use on_setattr &quot;</span>
                        <span class="s4">&quot;(frozen-ness was inherited).&quot;</span>
                    <span class="s1">)</span>

                <span class="s1">on_setattr = setters.NO_OP</span>
                <span class="s3">break</span>

        <span class="s3">if </span><span class="s1">auto_attribs </span><span class="s3">is not None</span><span class="s1">:</span>
            <span class="s3">return </span><span class="s1">do_it(cls</span><span class="s3">, </span><span class="s1">auto_attribs)</span>

        <span class="s3">try</span><span class="s1">:</span>
            <span class="s3">return </span><span class="s1">do_it(cls</span><span class="s3">, True</span><span class="s1">)</span>
        <span class="s3">except </span><span class="s1">UnannotatedAttributeError:</span>
            <span class="s3">return </span><span class="s1">do_it(cls</span><span class="s3">, False</span><span class="s1">)</span>

    <span class="s0"># maybe_cls's type depends on the usage of the decorator.  It's a class</span>
    <span class="s0"># if it's used as `@attrs` but ``None`` if used as `@attrs()`.</span>
    <span class="s3">if </span><span class="s1">maybe_cls </span><span class="s3">is None</span><span class="s1">:</span>
        <span class="s3">return </span><span class="s1">wrap</span>
    <span class="s3">else</span><span class="s1">:</span>
        <span class="s3">return </span><span class="s1">wrap(maybe_cls)</span>
