<span class="s2"># Notes on _infer_* functions.</span>
<span class="s2"># Unfortunately ``get_default_scheme()`` didn't exist before 3.10, so there's no</span>
<span class="s2"># way to ask things like &quot;what is the '_prefix' scheme on this platform&quot;. These</span>
<span class="s2"># functions try to answer that with some heuristics while accounting for ad-hoc</span>
<span class="s2"># platforms not covered by CPython's default sysconfig implementation. If the</span>
<span class="s2"># ad-hoc implementation does not fully implement sysconfig, we'll fall back to</span>
<span class="s2"># a POSIX scheme.</span>

<span class="s1">_AVAILABLE_SCHEMES = set(sysconfig.get_scheme_names())</span>

<span class="s1">_PREFERRED_SCHEME_API = getattr(sysconfig</span><span class="s0">, </span><span class="s3">&quot;get_preferred_scheme&quot;</span><span class="s0">, None</span><span class="s1">)</span>


<span class="s0">def </span><span class="s1">_should_use_osx_framework_prefix() -&gt; bool:</span>
    <span class="s4">&quot;&quot;&quot;Check for Apple's ``osx_framework_library`` scheme. 
 
    Python distributed by Apple's Command Line Tools has this special scheme 
    that's used when: 
 
    * This is a framework build. 
    * We are installing into the system prefix. 
 
    This does not account for ``pip install --prefix`` (also means we're not 
    installing to the system prefix), which should use ``posix_prefix``, but 
    logic here means ``_infer_prefix()`` outputs ``osx_framework_library``. But 
    since ``prefix`` is not available for ``sysconfig.get_default_scheme()``, 
    which is the stdlib replacement for ``_infer_prefix()``, presumably Apple 
    wouldn't be able to magically switch between ``osx_framework_library`` and 
    ``posix_prefix``. ``_infer_prefix()`` returning ``osx_framework_library`` 
    means its behavior is consistent whether we use the stdlib implementation 
    or our own, and we deal with this special case in ``get_scheme()`` instead. 
    &quot;&quot;&quot;</span>
    <span class="s0">return </span><span class="s1">(</span>
        <span class="s3">&quot;osx_framework_library&quot; </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES</span>
        <span class="s0">and not </span><span class="s1">running_under_virtualenv()</span>
        <span class="s0">and </span><span class="s1">is_osx_framework()</span>
    <span class="s1">)</span>


<span class="s0">def </span><span class="s1">_infer_prefix() -&gt; str:</span>
    <span class="s4">&quot;&quot;&quot;Try to find a prefix scheme for the current platform. 
 
    This tries: 
 
    * A special ``osx_framework_library`` for Python distributed by Apple's 
      Command Line Tools, when not running in a virtual environment. 
    * Implementation + OS, used by PyPy on Windows (``pypy_nt``). 
    * Implementation without OS, used by PyPy on POSIX (``pypy``). 
    * OS + &quot;prefix&quot;, used by CPython on POSIX (``posix_prefix``). 
    * Just the OS name, used by CPython on Windows (``nt``). 
 
    If none of the above works, fall back to ``posix_prefix``. 
    &quot;&quot;&quot;</span>
    <span class="s0">if </span><span class="s1">_PREFERRED_SCHEME_API:</span>
        <span class="s0">return </span><span class="s1">_PREFERRED_SCHEME_API(</span><span class="s3">&quot;prefix&quot;</span><span class="s1">)</span>
    <span class="s0">if </span><span class="s1">_should_use_osx_framework_prefix():</span>
        <span class="s0">return </span><span class="s3">&quot;osx_framework_library&quot;</span>
    <span class="s1">implementation_suffixed = </span><span class="s3">f&quot;</span><span class="s0">{</span><span class="s1">sys.implementation.name</span><span class="s0">}</span><span class="s3">_</span><span class="s0">{</span><span class="s1">os.name</span><span class="s0">}</span><span class="s3">&quot;</span>
    <span class="s0">if </span><span class="s1">implementation_suffixed </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>
        <span class="s0">return </span><span class="s1">implementation_suffixed</span>
    <span class="s0">if </span><span class="s1">sys.implementation.name </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>
        <span class="s0">return </span><span class="s1">sys.implementation.name</span>
    <span class="s1">suffixed = </span><span class="s3">f&quot;</span><span class="s0">{</span><span class="s1">os.name</span><span class="s0">}</span><span class="s3">_prefix&quot;</span>
    <span class="s0">if </span><span class="s1">suffixed </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>
        <span class="s0">return </span><span class="s1">suffixed</span>
    <span class="s0">if </span><span class="s1">os.name </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:  </span><span class="s2"># On Windows, prefx is just called &quot;nt&quot;.</span>
        <span class="s0">return </span><span class="s1">os.name</span>
    <span class="s0">return </span><span class="s3">&quot;posix_prefix&quot;</span>


<span class="s0">def </span><span class="s1">_infer_user() -&gt; str:</span>
    <span class="s4">&quot;&quot;&quot;Try to find a user scheme for the current platform.&quot;&quot;&quot;</span>
    <span class="s0">if </span><span class="s1">_PREFERRED_SCHEME_API:</span>
        <span class="s0">return </span><span class="s1">_PREFERRED_SCHEME_API(</span><span class="s3">&quot;user&quot;</span><span class="s1">)</span>
    <span class="s0">if </span><span class="s1">is_osx_framework() </span><span class="s0">and not </span><span class="s1">running_under_virtualenv():</span>

    <span class="s0">else</span><span class="s1">:</span>

    <span class="s0">if </span><span class="s1">suffixed </span><span class="s0">in </span><span class="s1">_AVAILABLE_SCHEMES:</span>

    <span class="s0">if </span><span class="s3">&quot;posix_user&quot; </span><span class="s0">not in </span><span class="s1">_AVAILABLE_SCHEMES:  </span><span class="s2"># User scheme unavailable.</span>

<span class="s0">def </span><span class="s1">_infer_home() -&gt; str:</span>
    <span class="s4">&quot;&quot;&quot;Try to find a home for the current platform.&quot;&quot;&quot;</span>

    &quot;&quot;&quot;</span>
