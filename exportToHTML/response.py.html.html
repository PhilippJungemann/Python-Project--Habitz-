<span class="s0">def </span><span class="s1">is_fp_closed(obj):</span>
    <span class="s2">&quot;&quot;&quot; 
    Checks whether a given file-like object is closed. 
 
    :param obj: 
        The file-like object to check. 
    &quot;&quot;&quot;</span>

    <span class="s0">try</span><span class="s1">:</span>
        <span class="s3"># Check `isclosed()` first, in case Python3 doesn't set `closed`.</span>

        <span class="s0">return </span><span class="s1">obj.isclosed()</span>
    <span class="s0">except </span><span class="s1">AttributeError:</span>
        <span class="s0">pass</span>

        <span class="s0">return </span><span class="s1">obj.closed</span>

        <span class="s0">pass</span>

    <span class="s0">except </span><span class="s1">AttributeError:</span>

<span class="s0">def </span><span class="s1">assert_header_parsing(headers):</span>
    <span class="s2">&quot;&quot;&quot; 
    Asserts whether all headers have been successfully parsed. 
    Extracts encountered errors from the result of parsing headers. 
 
    Only works on Python 3. 
 
    :param http.client.HTTPMessage headers: Headers to verify. 
 
    :raises urllib3.exceptions.HeaderParsingError: 
        If parsing errors are found. 
    &quot;&quot;&quot;</span>

    <span class="s3"># This will fail silently if we pass in the wrong kind of parameter.</span>

    <span class="s0">if not </span><span class="s1">isinstance(headers</span><span class="s0">, </span><span class="s1">httplib.HTTPMessage):</span>
        <span class="s0">raise </span><span class="s1">TypeError(</span><span class="s4">&quot;expected httplib.Message, got {0}.&quot;</span><span class="s1">.format(type(headers)))</span>

    <span class="s1">defects = getattr(headers</span><span class="s0">, </span><span class="s4">&quot;defects&quot;</span><span class="s0">, None</span><span class="s1">)</span>
    <span class="s1">get_payload = getattr(headers</span><span class="s0">, </span><span class="s4">&quot;get_payload&quot;</span><span class="s0">, None</span><span class="s1">)</span>

    <span class="s1">unparsed_data = </span><span class="s0">None</span>
    <span class="s0">if </span><span class="s1">get_payload:</span>
        <span class="s3"># get_payload is actually email.message.Message.get_payload;</span>
        <span class="s3"># we're only interested in the result if it's not a multipart message</span>
        <span class="s0">if not </span><span class="s1">headers.is_multipart():</span>
            <span class="s1">payload = get_payload()</span>

            <span class="s0">if </span><span class="s1">isinstance(payload</span><span class="s0">, </span><span class="s1">(bytes</span><span class="s0">, </span><span class="s1">str)):</span>
                <span class="s1">unparsed_data = payload</span>
    <span class="s0">if </span><span class="s1">defects:</span>
        <span class="s3"># httplib is assuming a response body is available</span>
        <span class="s3"># when parsing headers even when httplib only sends</span>
        <span class="s3"># header data to parse_headers() This results in</span>
        <span class="s3"># defects on multipart responses in particular.</span>
        <span class="s3"># See: https://github.com/urllib3/urllib3/issues/800</span>

        <span class="s3"># So we ignore the following defects:</span>
        <span class="s3"># - StartBoundaryNotFoundDefect:</span>
        <span class="s3">#     The claimed start boundary was never found.</span>
        <span class="s3"># - MultipartInvariantViolationDefect:</span>

        <span class="s1">defects = [</span>
            <span class="s1">defect</span>
            <span class="s0">for </span><span class="s1">defect </span><span class="s0">in </span><span class="s1">defects</span>

            <span class="s1">)</span>
        <span class="s1">]</span>

    <span class="s0">if </span><span class="s1">defects </span><span class="s0">or </span><span class="s1">unparsed_data:</span>
        <span class="s0">raise </span><span class="s1">HeaderParsingError(defects=defects</span><span class="s0">, </span><span class="s1">unparsed_data=unparsed_data)</span>
