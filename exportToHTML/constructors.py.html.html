<span class="s2">def </span><span class="s1">parse_editable(editable_req: str) -&gt; Tuple[Optional[str]</span><span class="s2">, </span><span class="s1">str</span><span class="s2">, </span><span class="s1">Set[str]]:</span>

    <span class="s1">url = editable_req</span>

    <span class="s5"># If a file path is specified with extras, strip off the extras.</span>
    <span class="s1">url_no_extras</span><span class="s2">, </span><span class="s1">extras = _strip_extras(url)</span>

    <span class="s2">if </span><span class="s1">os.path.isdir(url_no_extras):</span>

        <span class="s1">url_no_extras = path_to_url(url_no_extras)</span>

    <span class="s2">if </span><span class="s1">url_no_extras.lower().startswith(</span><span class="s3">&quot;file:&quot;</span><span class="s1">):</span>
        <span class="s1">package_name = Link(url_no_extras).egg_fragment</span>
        <span class="s2">if </span><span class="s1">extras:</span>
            <span class="s2">return </span><span class="s1">(</span>
                <span class="s1">package_name</span><span class="s2">,</span>
                <span class="s1">url_no_extras</span><span class="s2">,</span>
                <span class="s1">get_requirement(</span><span class="s3">&quot;placeholder&quot; </span><span class="s1">+ extras.lower()).extras</span><span class="s2">,</span>
            <span class="s1">)</span>
        <span class="s2">else</span><span class="s1">:</span>
            <span class="s2">return </span><span class="s1">package_name</span><span class="s2">, </span><span class="s1">url_no_extras</span><span class="s2">, </span><span class="s1">set()</span>

    <span class="s2">for </span><span class="s1">version_control </span><span class="s2">in </span><span class="s1">vcs:</span>
        <span class="s2">if </span><span class="s1">url.lower().startswith(</span><span class="s3">f&quot;</span><span class="s2">{</span><span class="s1">version_control</span><span class="s2">}</span><span class="s3">:&quot;</span><span class="s1">):</span>
            <span class="s1">url = </span><span class="s3">f&quot;</span><span class="s2">{</span><span class="s1">version_control</span><span class="s2">}</span><span class="s3">+</span><span class="s2">{</span><span class="s1">url</span><span class="s2">}</span><span class="s3">&quot;</span>
            <span class="s2">break</span>

    <span class="s1">link = Link(url)</span>

    <span class="s2">if not </span><span class="s1">link.is_vcs:</span>
        <span class="s1">backends = </span><span class="s3">&quot;, &quot;</span><span class="s1">.join(vcs.all_schemes)</span>
        <span class="s2">raise </span><span class="s1">InstallationError(</span>
            <span class="s3">f&quot;</span><span class="s2">{</span><span class="s1">editable_req</span><span class="s2">} </span><span class="s3">is not a valid editable requirement. &quot;</span>
            <span class="s3">f&quot;It should either be a path to a local project or a VCS URL &quot;</span>
            <span class="s3">f&quot;(beginning with </span><span class="s2">{</span><span class="s1">backends</span><span class="s2">}</span><span class="s3">).&quot;</span>

    <span class="s1">package_name = link.egg_fragment</span>
    <span class="s2">if not </span><span class="s1">package_name:</span>

<span class="s2">def </span><span class="s1">check_first_requirement_in_file(filename: str) -&gt; </span><span class="s2">None</span><span class="s1">:</span>
    <span class="s0">&quot;&quot;&quot;Check if file is parsable as a requirements file. 
 
    This is heavily based on ``pkg_resources.parse_requirements``, but 
    simplified to just check the first meaningful line. 
 
    :raises InvalidRequirement: If the first meaningful line cannot be parsed 
        as an requirement. 
    &quot;&quot;&quot;</span>
    <span class="s2">with </span><span class="s1">open(filename</span><span class="s2">, </span><span class="s1">encoding=</span><span class="s3">&quot;utf-8&quot;</span><span class="s2">, </span><span class="s1">errors=</span><span class="s3">&quot;ignore&quot;</span><span class="s1">) </span><span class="s2">as </span><span class="s1">f:</span>
        <span class="s5"># Create a steppable iterator, so we can handle \-continuations.</span>
        <span class="s1">lines = (</span>
            <span class="s1">line</span>
            <span class="s2">for </span><span class="s1">line </span><span class="s2">in </span><span class="s1">(line.strip() </span><span class="s2">for </span><span class="s1">line </span><span class="s2">in </span><span class="s1">f)</span>
            <span class="s2">if </span><span class="s1">line </span><span class="s2">and not </span><span class="s1">line.startswith(</span><span class="s3">&quot;#&quot;</span><span class="s1">)  </span><span class="s5"># Skip blank lines/comments.</span>
        <span class="s1">)</span>

        <span class="s2">for </span><span class="s1">line </span><span class="s2">in </span><span class="s1">lines:</span>
            <span class="s5"># Drop comments -- a hash without a space may be in a URL.</span>
            <span class="s2">if </span><span class="s3">&quot; #&quot; </span><span class="s2">in </span><span class="s1">line:</span>
                <span class="s1">line = line[: line.find(</span><span class="s3">&quot; #&quot;</span><span class="s1">)]</span>
            <span class="s5"># If there is a line continuation, drop it, and append the next line.</span>
            <span class="s2">if </span><span class="s1">line.endswith(</span><span class="s3">&quot;</span><span class="s2">\\</span><span class="s3">&quot;</span><span class="s1">):</span>
                <span class="s1">line = line[:-</span><span class="s4">2</span><span class="s1">].strip() + next(lines</span><span class="s2">, </span><span class="s3">&quot;&quot;</span><span class="s1">)</span>
            <span class="s1">Requirement(line)</span>
            <span class="s2">return</span>
